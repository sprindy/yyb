71 static void led_softblink_on_timeout(void *p_context)
72 {
73  uint32_t new_timeout_ticks;
74  uint32_t err_code;
75 
76  VERIFY_MODULE_INITIALIZED_VOID();
77 
78  if (m_led_sb.leds_is_on)
79  {
80  // Turn off LEDs
81  if (m_led_sb.params.active_high)
82  {
83  NRF_GPIO->OUTCLR = m_led_sb.bit_mask;
84  }
85  else
86  {
87  NRF_GPIO->OUTSET = m_led_sb.bit_mask;
88  }
89 
90  m_led_sb.leds_is_on = false;
91 
92  if (m_led_sb.is_counting_up)
93  {
94  m_led_sb.duty_cycle += m_led_sb.params.duty_cycle_step;
95  if(m_led_sb.duty_cycle >= m_led_sb.params.duty_cycle_max)
96  {
97  // Max PWM duty cycle is reached, start decrementing.
98  m_led_sb.is_counting_up = false;
99  }
100  }
101  else
102  {
103  m_led_sb.duty_cycle -= m_led_sb.params.duty_cycle_step;
104  if(m_led_sb.duty_cycle <= m_led_sb.params.duty_cycle_min)
105  {
106  // Min PWM duty cycle is reached, start incrementing.
107  m_led_sb.is_counting_up = true;
108  }
109  }
110 
111  if (m_led_sb.duty_cycle <= m_led_sb.params.duty_cycle_min)
112  {
113  new_timeout_ticks = APP_TIMER_TICKS(m_led_sb.params.off_time_ms, APP_TIMER_PRESCALER);
114  }
115  else
116  {
117  new_timeout_ticks = (LED_PWM_PERIOD_TICKS / 100) * (100 - m_led_sb.duty_cycle) + 5;
118  }
119  }
120  else
121  {
122  // Turn on LEDs
123  if (m_led_sb.params.active_high)
124  {
125  NRF_GPIO->OUTSET = m_led_sb.bit_mask;;
126  }
127  else
128  {
129  NRF_GPIO->OUTCLR = m_led_sb.bit_mask;;
130  }
131 
132  m_led_sb.leds_is_on = true;
133 
134  new_timeout_ticks = (LED_PWM_PERIOD_TICKS / 100) * m_led_sb.duty_cycle + 5;
135  }
136 
137  err_code = app_timer_start(m_led_sb.timer_id, new_timeout_ticks, NULL);
138  APP_ERROR_CHECK(err_code);
139 }

194 uint32_t led_softblink_init(led_sb_init_params_t *init_params)
195 {
196  uint32_t bit_mask;
197  uint32_t pin_number = 0;
198 
199  if (m_led_sb.initialized)
200  {
201  return NRF_ERROR_INVALID_STATE;
202  }
203 
204  NULL_PARAM_CHECK(init_params);
205 
206  if ( (init_params->duty_cycle_max == 0) ||
207  (init_params->duty_cycle_max <= init_params->duty_cycle_min) ||
208  (init_params->duty_cycle_step == 0) ||
209  (init_params->leds_pin_bm == 0))
210  {
211  return NRF_ERROR_INVALID_PARAM;
212  }
213  memset(&m_led_sb, 0, sizeof(led_sb_context_t));
214  memcpy(&m_led_sb.params, init_params, sizeof(led_sb_init_params_t));
215  m_led_sb.is_counting_up = true;
216 
217  bit_mask = m_led_sb.params.leds_pin_bm;
218  // Configure IO according to the leds_pin_bit_mask
219  while (bit_mask)
220  {
221  if (bit_mask & 0x1UL)
222  {
223  nrf_gpio_cfg_output(pin_number);
224 
225  if (m_led_sb.params.active_high)
226  {
227  nrf_gpio_pin_clear(pin_number);
228  }
229  else
230  {
231  nrf_gpio_pin_set(pin_number);
232  }
233  }
234 
235  pin_number++;
236  bit_mask >>= 1UL;
237  }
238 
239  m_led_sb.initialized = true;
240 
241  return app_timer_create(&m_led_sb.timer_id, APP_TIMER_MODE_SINGLE_SHOT, led_softblink_on_timeout);
242 }
