58 static void pstorage_callback_handler(pstorage_handle_t * p_handle,
59  uint8_t op_code,
60  uint32_t result,
61  uint8_t * p_data,
62  uint32_t data_len)
63 {
64  switch (op_code)
65  {
66  case PSTORAGE_STORE_OP_CODE:
67  if ((m_dfu_state == DFU_STATE_RX_DATA_PKT) && (m_data_pkt_cb != NULL))
68  {
69  m_data_pkt_cb(DATA_PACKET, result, p_data);
70  }
71  break;
72 
73  case PSTORAGE_CLEAR_OP_CODE:
74  if (m_dfu_state == DFU_STATE_PREPARING)
75  {
76  m_functions.cleared();
77  m_dfu_state = DFU_STATE_RDY;
78  if (m_data_pkt_cb != NULL)
79  {
80  m_data_pkt_cb(START_PACKET, result, p_data);
81  }
82  }
83  break;
84 
85  default:
86  break;
87  }
88  APP_ERROR_CHECK(result);
89 }

168 uint32_t offset_calculate(uint32_t sd_image_size)
169 {
170  uint32_t offset = 0;
171 
172  if (m_start_packet.sd_image_size > DFU_BANK_0_REGION_START)
173  {
174  uint32_t page_mask = (CODE_PAGE_SIZE - 1);
175  uint32_t diff = m_start_packet.sd_image_size - DFU_BANK_0_REGION_START;
176 
177  offset = diff & ~page_mask;
178 
179  // Align offset to next page if image size is not page sized.
180  if ((diff & page_mask) > 0)
181  {
182  offset += CODE_PAGE_SIZE;
183  }
184  }
185 
186  return offset;
187 }

302 uint32_t dfu_start_pkt_handle(dfu_update_packet_t * p_packet)
303 {
304  uint32_t err_code;
305 
306  m_start_packet = *(p_packet->params.start_packet);
307 
308  // Check that the requested update procedure is supported.
309  // Currently the following combinations are allowed:
310  // - Application
311  // - SoftDevice
312  // - Bootloader
313  // - SoftDevice with Bootloader
314  if (IS_UPDATING_APP(m_start_packet) &&
315  (IS_UPDATING_SD(m_start_packet) || IS_UPDATING_BL(m_start_packet)))
316  {
317  // App update is only supported independently.
318  return NRF_ERROR_NOT_SUPPORTED;
319  }
320 
321  if (!(IS_WORD_SIZED(m_start_packet.sd_image_size) &&
322  IS_WORD_SIZED(m_start_packet.bl_image_size) &&
323  IS_WORD_SIZED(m_start_packet.app_image_size)))
324  {
325  // Image_sizes are not a multiple of 4 (word size).
326  return NRF_ERROR_NOT_SUPPORTED;
327  }
328 
329  m_image_size = m_start_packet.sd_image_size + m_start_packet.bl_image_size +
330  m_start_packet.app_image_size;
331 
332  if (m_start_packet.bl_image_size > DFU_BL_IMAGE_MAX_SIZE)
333  {
334  return NRF_ERROR_DATA_SIZE;
335  }
336 
337  if (m_image_size > (DFU_IMAGE_MAX_SIZE_FULL - CODE_PAGE_SIZE))
338  {
339  return NRF_ERROR_DATA_SIZE;
340  }
341  m_functions.prepare = dfu_prepare_func_app_erase;
342  m_functions.cleared = dfu_cleared_func_app;
343 
344  if (IS_UPDATING_SD(m_start_packet))
345  {
346  m_functions.activate = dfu_activate_sd;
347  }
348  else if (IS_UPDATING_BL(m_start_packet))
349  {
350  m_functions.activate = dfu_activate_bl;
351  }
352  else
353  {
354  m_functions.activate = dfu_activate_app;
355  }
356 
357  switch (m_dfu_state)
358  {
359  case DFU_STATE_IDLE:
360  // Valid peer activity detected. Hence restart the DFU timer.
361  err_code = dfu_timer_restart();
362  if (err_code != NRF_SUCCESS)
363  {
364  return err_code;
365  }
366  m_functions.prepare(m_image_size);
367 
368  break;
369 
370  default:
371  err_code = NRF_ERROR_INVALID_STATE;
372  break;
373  }
374 
375  return err_code;
376 }

379 uint32_t dfu_data_pkt_handle(dfu_update_packet_t * p_packet)
380 {
381  uint32_t data_length;
382  uint32_t err_code;
383  uint32_t * p_data;
384 
385  if (p_packet == NULL)
386  {
387  return NRF_ERROR_NULL;
388  }
389 
390  // Check pointer alignment.
391  if (!is_word_aligned(p_packet->params.data_packet.p_data_packet))
392  {
393  // The p_data_packet is not word aligned address.
394  return NRF_ERROR_INVALID_ADDR;
395  }
396 
397  switch (m_dfu_state)
398  {
399  case DFU_STATE_RDY:
400  case DFU_STATE_RX_INIT_PKT:
401  return NRF_ERROR_INVALID_STATE;
402 
403  case DFU_STATE_RX_DATA_PKT:
404  data_length = p_packet->params.data_packet.packet_length * sizeof(uint32_t);
405 
406  if ((m_data_received + data_length) > m_image_size)
407  {
408  // The caller is trying to write more bytes into the flash than the size provided to
409  // the dfu_image_size_set function. This is treated as a serious error condition and
410  // an unrecoverable one. Hence point the variable mp_app_write_address to the top of
411  // the flash area. This will ensure that all future application data packet writes
412  // will be blocked because of the above check.
413  m_data_received = 0xFFFFFFFF;
414 
415  return NRF_ERROR_DATA_SIZE;
416  }
417 
418  // Valid peer activity detected. Hence restart the DFU timer.
419  err_code = dfu_timer_restart();
420  if (err_code != NRF_SUCCESS)
421  {
422  return err_code;
423  }
424 
425  p_data = (uint32_t *)p_packet->params.data_packet.p_data_packet;
426 
427  err_code = pstorage_raw_store(mp_storage_handle_active,
428  (uint8_t *)p_data,
429  data_length,
430  m_data_received);
431  if (err_code != NRF_SUCCESS)
432  {
433  return err_code;
434  }
435 
436  m_data_received += data_length;
437 
438  if (m_data_received != m_image_size)
439  {
440  // The entire image is not received yet. More data is expected.
441  err_code = NRF_ERROR_INVALID_LENGTH;
442  }
443  else
444  {
445  // The entire image has been received. Return NRF_SUCCESS.
446  err_code = NRF_SUCCESS;
447  }
448  break;
449 
450  default:
451  err_code = NRF_ERROR_INVALID_STATE;
452  break;
453  }
454 
455  return err_code;
456 }

459 uint32_t dfu_init_pkt_complete(void)
460 {
461  uint32_t err_code = NRF_ERROR_INVALID_STATE;
462 
463  // DFU initialization has been done and a start packet has been received.
464  if (IMAGE_WRITE_IN_PROGRESS())
465  {
466  // Image write is already in progress. Cannot handle an init packet now.
467  return NRF_ERROR_INVALID_STATE;
468  }
469 
470  if (m_dfu_state == DFU_STATE_RX_INIT_PKT)
471  {
472  err_code = dfu_init_prevalidate(m_init_packet, m_init_packet_length);
473  if (err_code == NRF_SUCCESS)
474  {
475  m_dfu_state = DFU_STATE_RX_DATA_PKT;
476  }
477  else
478  {
479  m_init_packet_length = 0;
480  }
481  }
482  return err_code;

486 uint32_t dfu_init_pkt_handle(dfu_update_packet_t * p_packet)
487 {
488  uint32_t err_code = NRF_SUCCESS;
489  uint32_t length;
490 
491  switch (m_dfu_state)
492  {
493  case DFU_STATE_RDY:
494  m_dfu_state = DFU_STATE_RX_INIT_PKT;
495  // When receiving init packet in state ready just update and fall through this case.
496 
497  case DFU_STATE_RX_INIT_PKT:
498  // DFU initialization has been done and a start packet has been received.
499  if (IMAGE_WRITE_IN_PROGRESS())
500  {
501  // Image write is already in progress. Cannot handle an init packet now.
502  return NRF_ERROR_INVALID_STATE;
503  }
504 
505  // Valid peer activity detected. Hence restart the DFU timer.
506  err_code = dfu_timer_restart();
507  if (err_code != NRF_SUCCESS)
508  {
509  return err_code;
510  }
511 
512  length = p_packet->params.data_packet.packet_length * sizeof(uint32_t);
513  if ((m_init_packet_length + length) > sizeof(m_init_packet))
514  {
515  return NRF_ERROR_INVALID_LENGTH;
516  }
517 
518  memcpy(&m_init_packet[m_init_packet_length],
519  &p_packet->params.data_packet.p_data_packet[0],
520  length);
521  m_init_packet_length += length;
522  break;
523 
524  default:
525  // Either the start packet was not received or dfu_init function was not called before.
526  err_code = NRF_ERROR_INVALID_STATE;
527  break;
528  }
529 
530  return err_code;
531 }

534 uint32_t dfu_image_validate()
535 {
536  uint32_t err_code;
537 
538  switch (m_dfu_state)
539  {
540  case DFU_STATE_RX_DATA_PKT:
541  m_dfu_state = DFU_STATE_VALIDATE;
542 
543  // Check if the application image write has finished.
544  if (m_data_received != m_image_size)
545  {
546  // Image not yet fully transfered by the peer or the peer has attempted to write
547  // too much data. Hence the validation should fail.
548  err_code = NRF_ERROR_INVALID_STATE;
549  }
550  else
551  {
552  // Valid peer activity detected. Hence restart the DFU timer.
553  err_code = dfu_timer_restart();

555  {
556  err_code = dfu_init_postvalidate((uint8_t *)mp_storage_handle_active->block_id,
557  m_image_size);
558  if (err_code != NRF_SUCCESS)
559  {
560  return err_code;
561  }
562 
563  m_dfu_state = DFU_STATE_WAIT_4_ACTIVATE;
564  }
565  }
566  break;
567 
568  default:
569  err_code = NRF_ERROR_INVALID_STATE;
570  break;
571  }
572 
573  return err_code;

637 static uint32_t dfu_sd_img_block_swap(uint32_t * src,
638  uint32_t * dst,
639  uint32_t len,
640  uint32_t block_size)
641 {
642  // It is neccesarry to swap the new SoftDevice in 3 rounds to ensure correct copy of data
643  // and verifucation of data in case power reset occurs during write to flash.
644  // To ensure the robustness of swapping the images are compared backwards till start of
645  // image swap. If the back is identical everything is swapped.
646  uint32_t err_code = dfu_compare_block(src, dst, len);
647  if (err_code == NRF_SUCCESS)
648  {
649  return err_code;
650  }
651 
652  if ((uint32_t)dst > SOFTDEVICE_REGION_START)
653  {
654  err_code = dfu_sd_img_block_swap((uint32_t *)((uint32_t)src - block_size),
655  (uint32_t *)((uint32_t)dst - block_size),
656  block_size,
657  block_size);
658  if (err_code != NRF_SUCCESS)
659  {
660  return err_code;
661  }
662  }
663 
664  err_code = dfu_copy_sd(src, dst, len);
665  if (err_code != NRF_SUCCESS)
666  {
667  return err_code;
668  }
669  return dfu_compare_block(src, dst, len);
670 }

673 uint32_t dfu_sd_image_swap(void)
674 {
675  bootloader_settings_t boot_settings;
676 
677  bootloader_settings_get(&boot_settings);
678 
679  if (boot_settings.sd_image_size == 0)
680  {
681  return NRF_SUCCESS;
682  }
683 
684  if ((SOFTDEVICE_REGION_START + boot_settings.sd_image_size) > boot_settings.sd_image_start)
685  {
686  uint32_t err_code;
687  uint32_t sd_start = SOFTDEVICE_REGION_START;
688  uint32_t block_size = (boot_settings.sd_image_start - sd_start) / 2;
689  uint32_t image_end = boot_settings.sd_image_start + boot_settings.sd_image_size;
690 
691  uint32_t img_block_start = boot_settings.sd_image_start + 2 * block_size;
692  uint32_t sd_block_start = sd_start + 2 * block_size;
693 
694  if (SOFTDEVICE_INFORMATION->softdevice_size < boot_settings.sd_image_size)
695  {
696  // This will clear a page thus ensuring the old image is invalidated before swapping.
697  err_code = dfu_copy_sd((uint32_t *)(sd_start + block_size),
698  (uint32_t *)(sd_start + block_size),
699  sizeof(uint32_t));
700  if (err_code != NRF_SUCCESS)
701  {
702  return err_code;
703  }
704 
705  err_code = dfu_copy_sd((uint32_t *)sd_start, (uint32_t *)sd_start, sizeof(uint32_t));
706  if (err_code != NRF_SUCCESS)
707  {
708  return err_code;
709  }
710  }
711 
712  return dfu_sd_img_block_swap((uint32_t *)img_block_start,
713  (uint32_t *)sd_block_start,
714  image_end - img_block_start,
715  block_size);
716  }
717  else
718  {
719  if (boot_settings.sd_image_size != 0)
720  {
721  return dfu_copy_sd((uint32_t *)boot_settings.sd_image_start,
722  (uint32_t *)SOFTDEVICE_REGION_START,
723  boot_settings.sd_image_size);
724  }
725  }
726 
727  return NRF_SUCCESS;
728 }

803  {

810  block_size);
