115 static ble_dfu_resp_val_t nrf_err_code_translate(uint32_t err_code,
116  const ble_dfu_procedure_t current_dfu_proc)
117 {
118  switch (err_code)
119  {
120  case NRF_SUCCESS:
121  return BLE_DFU_RESP_VAL_SUCCESS;
122 
123  case NRF_ERROR_INVALID_STATE:
124  return BLE_DFU_RESP_VAL_INVALID_STATE;
125 
126  case NRF_ERROR_NOT_SUPPORTED:

154 static void dfu_cb_handler(uint32_t packet, uint32_t result, uint8_t * p_data)
155 {
156  switch (packet)
157  {
158  ble_dfu_resp_val_t resp_val;
159  uint32_t err_code;
160 
161  case DATA_PACKET:
162  if (result != NRF_SUCCESS)
163  {
164  // Disconnect from peer.
165  if (IS_CONNECTED())

167  err_code = sd_ble_gap_disconnect(m_conn_handle,

222 static void start_data_process(ble_dfu_t * p_dfu, ble_dfu_evt_t * p_evt)
223 {
224  uint32_t err_code;
225 
226  dfu_start_packet_t start_packet = {.dfu_update_mode = m_update_mode};
227  dfu_update_packet_t update_packet =
228  {
229  .packet_type = START_PACKET,
230  .params.start_packet = &start_packet
231  };
232 
233  uint32_t length = p_evt->evt.ble_dfu_pkt_write.len;
234 
235  // Verify that the data is exactly three * four bytes (three words) long.
236  // Or a single word to support backwards compatibility of application image transfer.
237  if (length != (3 * sizeof(uint32_t)))
238  {
239  err_code = ble_dfu_response_send(p_dfu,
240  BLE_DFU_START_PROCEDURE,
241  BLE_DFU_RESP_VAL_NOT_SUPPORTED);
242  APP_ERROR_CHECK(err_code);
243  }
244  else
245  {
246  // Extract the size of from the DFU Packet Characteristic.
247  uint8_t * length_data = p_evt->evt.ble_dfu_pkt_write.p_data;
248 
249  start_packet.sd_image_size = uint32_decode(length_data);
250  start_packet.bl_image_size = uint32_decode(length_data + 4);
251  start_packet.app_image_size = uint32_decode(length_data + 8);
252 
253  err_code = dfu_start_pkt_handle(&update_packet);
254  if (err_code != NRF_SUCCESS)
255  {
256  // Translate the err_code returned by the above function to DFU Response Value.
257  ble_dfu_resp_val_t resp_val;
258 
259  resp_val = nrf_err_code_translate(err_code, BLE_DFU_START_PROCEDURE);
260 
261  err_code = ble_dfu_response_send(p_dfu, BLE_DFU_START_PROCEDURE, resp_val);
262  }
263 
264  APP_ERROR_CHECK(err_code);
265  }
266 }

275 static void init_data_process(ble_dfu_t * p_dfu, ble_dfu_evt_t * p_evt)
276 {
277  uint32_t err_code;
278  dfu_update_packet_t dfu_pkt;
279  uint8_t num_of_padding_bytes = 0;
280 
281  dfu_pkt.packet_type = INIT_PACKET;
282 
283  dfu_pkt.params.data_packet.p_data_packet = (uint32_t *)p_evt->evt.ble_dfu_pkt_write.p_data;
284 
285  // The DFU module accepts the dfu_pkt.packet_length to be in 'number of words'. And so if the
286  // received data does not have a size which is a multiple of four, it should be padded with
287  // zeros and the packet_length should be incremented accordingly before calling
288  // dfu_init_pkt_handle.
289  if ((p_evt->evt.ble_dfu_pkt_write.len & (sizeof(uint32_t) - 1)) != 0)
290  {
291  // Find out the number of bytes to be padded.
292  num_of_padding_bytes = sizeof(uint32_t)
293  -
294  (p_evt->evt.ble_dfu_pkt_write.len & (sizeof(uint32_t) - 1));
295 
296  uint32_t i;
297  for (i = 0; i < num_of_padding_bytes; i++)
298  {
299  dfu_pkt.params.data_packet.p_data_packet[p_evt->evt.ble_dfu_pkt_write.len + i] = 0;
300  }
301  }
302 
303  dfu_pkt.params.data_packet.packet_length =
304  (p_evt->evt.ble_dfu_pkt_write.len + num_of_padding_bytes) / sizeof(uint32_t);
305 
306  err_code = dfu_init_pkt_handle(&dfu_pkt);
307 
308  // Translate the err_code returned by the above function to DFU Response Value.
309  if (err_code != NRF_SUCCESS)
310  {
311  ble_dfu_resp_val_t resp_val = nrf_err_code_translate(err_code, BLE_DFU_INIT_PROCEDURE);
312 
313  err_code = ble_dfu_response_send(p_dfu, BLE_DFU_INIT_PROCEDURE, resp_val);
314  APP_ERROR_CHECK(err_code);
315  }
316 }

325 static void app_data_process(ble_dfu_t * p_dfu, ble_dfu_evt_t * p_evt)
326 {
327  uint32_t err_code;
328 
329  if ((p_evt->evt.ble_dfu_pkt_write.len & (sizeof(uint32_t) - 1)) != 0)
330  {
331  // Data length is not a multiple of 4 (word size).
332  err_code = ble_dfu_response_send(p_dfu,
333  BLE_DFU_RECEIVE_APP_PROCEDURE,
334  BLE_DFU_RESP_VAL_NOT_SUPPORTED);
335  APP_ERROR_CHECK(err_code);
336  return;
337  }
338 
339  uint32_t length = p_evt->evt.ble_dfu_pkt_write.len;
340 
341  err_code = hci_mem_pool_rx_produce(length, (void**) &mp_rx_buffer);
342  if (err_code != NRF_SUCCESS)
343  {
344  dfu_error_notify(p_dfu, err_code);
345  return;
346  }
347 
348  uint8_t * p_data_packet = p_evt->evt.ble_dfu_pkt_write.p_data;
349  memcpy(mp_rx_buffer, p_data_packet, length);
350 
351  err_code = hci_mem_pool_rx_data_size_set(length);
352  if (err_code != NRF_SUCCESS)
353  {
354  dfu_error_notify(p_dfu, err_code);
355  return;
356  }
357 
358  err_code = hci_mem_pool_rx_extract(&mp_rx_buffer, &length);
359  if (err_code != NRF_SUCCESS)
360  {
361  dfu_error_notify(p_dfu, err_code);
362  return;
363  }
364 
365  dfu_update_packet_t dfu_pkt;
366 
367  dfu_pkt.packet_type = DATA_PACKET;
368  dfu_pkt.params.data_packet.packet_length = length / sizeof(uint32_t);
369  dfu_pkt.params.data_packet.p_data_packet = (uint32_t*)mp_rx_buffer;
370 
371  err_code = dfu_data_pkt_handle(&dfu_pkt);
372 
373  if (err_code == NRF_SUCCESS)
374  {
375  // All the expected firmware data has been received and processed successfully.
376 
377  m_num_of_firmware_bytes_rcvd += p_evt->evt.ble_dfu_pkt_write.len;
378 
379  // Notify the DFU Controller about the success about the procedure.
380  err_code = ble_dfu_response_send(p_dfu,
381  BLE_DFU_RECEIVE_APP_PROCEDURE,
382  BLE_DFU_RESP_VAL_SUCCESS);
383  APP_ERROR_CHECK(err_code);
384  }
385  else if (err_code == NRF_ERROR_INVALID_LENGTH)
386  {
387  // Firmware data packet was handled successfully. And more firmware data is expected.
388  m_num_of_firmware_bytes_rcvd += p_evt->evt.ble_dfu_pkt_write.len;
389 
390  // Check if a packet receipt notification is needed to be sent.
391  if (m_pkt_rcpt_notif_enabled)
392  {
393  // Decrement the counter for the number firmware packets needed for sending the
394  // next packet receipt notification.
395  m_pkt_notif_target_cnt--;
396 
397  if (m_pkt_notif_target_cnt == 0)
398  {
399  err_code = ble_dfu_pkts_rcpt_notify(p_dfu, m_num_of_firmware_bytes_rcvd);
400  APP_ERROR_CHECK(err_code);
401 
402  // Reset the counter for the number of firmware packets.
403  m_pkt_notif_target_cnt = m_pkt_notif_target;
404  }
405  }
406  }
407  else
408  {
409  uint32_t hci_error = hci_mem_pool_rx_consume(mp_rx_buffer);
410  if (hci_error != NRF_SUCCESS)
411  {
412  dfu_error_notify(p_dfu, hci_error);
413  }
414 
415  dfu_error_notify(p_dfu, err_code);
416  }
417 }

425 static void on_dfu_pkt_write(ble_dfu_t * p_dfu, ble_dfu_evt_t * p_evt)
426 {
427  // The peer has written to the DFU Packet characteristic. Depending on the value of
428  // the current value of the DFU Control Point, the appropriate action is taken.
429  switch (m_pkt_type)
430  {
431  case PKT_TYPE_START:
432  // The peer has written a start packet to the DFU Packet characteristic.
433  start_data_process(p_dfu, p_evt);
434  break;
435 
436  case PKT_TYPE_INIT:
437  // The peer has written an init packet to the DFU Packet characteristic.
438  init_data_process(p_dfu, p_evt);
439  break;
440 
441  case PKT_TYPE_FIRMWARE_DATA:
442  app_data_process(p_dfu, p_evt);
443  break;
444 
445  default:
446  // It is not possible to find out what packet it is. Ignore. Currently there is no
447  // mechanism to notify the DFU Controller about this error condition.
448  break;
449  }
450 }

494 static void on_dfu_evt(ble_dfu_t * p_dfu, ble_dfu_evt_t * p_evt)
495 {
496  uint32_t err_code;
497  ble_dfu_resp_val_t resp_val;
498 
499  switch (p_evt->ble_dfu_evt_type)
500  {
501  case BLE_DFU_VALIDATE:
502  err_code = dfu_image_validate();
503 
504  // Translate the err_code returned by the above function to DFU Response Value.
505  resp_val = nrf_err_code_translate(err_code, BLE_DFU_VALIDATE_PROCEDURE);
506 
507  err_code = ble_dfu_response_send(p_dfu, BLE_DFU_VALIDATE_PROCEDURE, resp_val);
508  APP_ERROR_CHECK(err_code);
509  break;
510 
511  case BLE_DFU_ACTIVATE_N_RESET:
512  err_code = dfu_transport_close();
513  APP_ERROR_CHECK(err_code);
514 
515  // With the S110 Flash API it is safe to initiate the activate before connection is
516  // fully closed.
517  err_code = dfu_image_activate();
518  if (err_code != NRF_SUCCESS)
519  {
520  dfu_reset();
521  }
522  break;
523 
524  case BLE_DFU_SYS_RESET:
525  err_code = dfu_transport_close();
526  APP_ERROR_CHECK(err_code);
527 
528  dfu_reset();
529  break;
530 
531  case BLE_DFU_START:
532  m_pkt_type = PKT_TYPE_START;
533  m_update_mode = (uint8_t)p_evt->evt.ble_dfu_pkt_write.p_data[0];
534  break;
535 
536  case BLE_DFU_RECEIVE_INIT_DATA:
537  m_pkt_type = PKT_TYPE_INIT;
538  if ((uint8_t)p_evt->evt.ble_dfu_pkt_write.p_data[0] == DFU_INIT_COMPLETE)
539  {
540  err_code = dfu_init_pkt_complete();
541 
542  // Translate the err_code returned by the above function to DFU Response Value.
543  resp_val = nrf_err_code_translate(err_code, BLE_DFU_INIT_PROCEDURE);
544 
545  err_code = ble_dfu_response_send(p_dfu, BLE_DFU_INIT_PROCEDURE, resp_val);
546  APP_ERROR_CHECK(err_code);
547  }
548  break;
549 
550  case BLE_DFU_RECEIVE_APP_DATA:
551  m_pkt_type = PKT_TYPE_FIRMWARE_DATA;
552  break;
553 
554  case BLE_DFU_PACKET_WRITE:
555  on_dfu_pkt_write(p_dfu, p_evt);
556  break;
557 
558  case BLE_DFU_PKT_RCPT_NOTIF_ENABLED:
559  m_pkt_rcpt_notif_enabled = true;
560  m_pkt_notif_target = p_evt->evt.pkt_rcpt_notif_req.num_of_pkts;
561  m_pkt_notif_target_cnt = p_evt->evt.pkt_rcpt_notif_req.num_of_pkts;
562  break;
563 
564  case BLE_DFU_PKT_RCPT_NOTIF_DISABLED:
565  m_pkt_rcpt_notif_enabled = false;
566  m_pkt_notif_target = 0;
567  break;
568 
569  case BLE_DFU_BYTES_RECEIVED_SEND:
570  err_code = ble_dfu_bytes_rcvd_report(p_dfu, m_num_of_firmware_bytes_rcvd);
571  APP_ERROR_CHECK(err_code);
572  break;
573 
574  default:
575  // Unsupported event received from DFU Service. Ignore.
576  break;
577  }
578 }

693 static void on_ble_evt(ble_evt_t * p_ble_evt)
694 {
695  uint32_t err_code;
696 
697  switch (p_ble_evt->header.evt_id)
698  {
699  case BLE_GAP_EVT_CONNECTED:
700  m_conn_handle = p_ble_evt->evt.gap_evt.conn_handle;
701  m_is_advertising = false;
702  break;
703 
704  case BLE_GAP_EVT_DISCONNECTED:
705  m_direct_adv_cnt = APP_DIRECTED_ADV_TIMEOUT;
706 
707  if (!m_tear_down_in_progress)
708  {
709  // The Disconnected event is because of an external event. (Link loss or
710  // disconnect triggered by the DFU Controller before the firmware update was
711  // complete).
712  // Restart advertising so that the DFU Controller can reconnect if possible.
713  advertising_start();
714  }
715 
716  m_conn_handle = BLE_CONN_HANDLE_INVALID;
717 
718  break;
719 
720  case BLE_GAP_EVT_SEC_PARAMS_REQUEST:
721  err_code = sd_ble_gap_sec_params_reply(m_conn_handle,
722  BLE_GAP_SEC_STATUS_SUCCESS,
723  &m_sec_params);
724  APP_ERROR_CHECK(err_code);
725  break;
726 
727  case BLE_GATTS_EVT_TIMEOUT:
728  if (p_ble_evt->evt.gatts_evt.params.timeout.src == BLE_GATT_TIMEOUT_SRC_PROTOCOL)
729  {
730 
731  err_code = sd_ble_gap_disconnect(m_conn_handle,
732  BLE_HCI_REMOTE_USER_TERMINATED_CONNECTION);
733  APP_ERROR_CHECK(err_code);
734  }
735  break;
736 
737  case BLE_GAP_EVT_TIMEOUT:
738  if (p_ble_evt->evt.gap_evt.params.timeout.src == BLE_GAP_TIMEOUT_SRC_ADVERTISEMENT)
739  {
740  m_is_advertising = false;
741  m_direct_adv_cnt--;
742  if (m_direct_adv_cnt == 0)
743  {
744  dfu_update_status_t update_status = {.status_code = DFU_TIMEOUT};
745  bootloader_dfu_update_process(update_status);
746  }
747  else
748  {
749  advertising_start();
750  }
751  }
752  break;
753 
754  case BLE_GAP_EVT_SEC_INFO_REQUEST:
755  err_code = sd_ble_gap_sec_info_reply(p_ble_evt->evt.gap_evt.conn_handle,
756  &m_ble_peer_data.enc_info,
757  NULL);
758  APP_ERROR_CHECK(err_code);
759  break;
760 
761  case BLE_GATTS_EVT_SYS_ATTR_MISSING:
762  case BLE_GAP_EVT_CONN_SEC_UPDATE:
763  err_code = sd_ble_gatts_sys_attr_set(m_conn_handle, NULL, 0);
764  APP_ERROR_CHECK(err_code);
765  break;
766 
767  default:
768  // No implementation needed.
769  break;
770  }
771 }

903 uint32_t dfu_transport_close()
904 {
