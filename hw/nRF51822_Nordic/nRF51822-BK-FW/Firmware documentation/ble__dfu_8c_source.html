255 static uint32_t on_ctrl_pt_write(ble_dfu_t * p_dfu, ble_gatts_evt_write_t * p_ble_write_evt)
256 {
257  ble_gatts_rw_authorize_reply_params_t write_authorize_reply;
258 
259  write_authorize_reply.type = BLE_GATTS_AUTHORIZE_TYPE_WRITE;
260 
261  if (!is_cccd_configured(p_dfu))
262  {
263  // Send an error response to the peer indicating that the CCCD is improperly configured.
264  write_authorize_reply.params.write.gatt_status =
265  BLE_GATT_STATUS_ATTERR_CPS_CCCD_CONFIG_ERROR;
266 
267  return (sd_ble_gatts_rw_authorize_reply(p_dfu->conn_handle, &write_authorize_reply));
268 
269  }
270  else
271  {
272  uint32_t err_code;
273 
274  write_authorize_reply.params.write.gatt_status = BLE_GATT_STATUS_SUCCESS;
275 
276  err_code = (sd_ble_gatts_rw_authorize_reply(p_dfu->conn_handle, &write_authorize_reply));
277 
278  if (err_code != NRF_SUCCESS)
279  {
280  return err_code;
281  }
282  }
283 
284  ble_dfu_evt_t ble_dfu_evt;
285 
286  switch (p_ble_write_evt->data[0])
287  {
288  case OP_CODE_START_DFU:
289  ble_dfu_evt.ble_dfu_evt_type = BLE_DFU_START;
290 
291  if (p_ble_write_evt->len < PKT_START_DFU_PARAM_LEN)
292  {
293  return ble_dfu_response_send(p_dfu,
294  (ble_dfu_procedure_t) p_ble_write_evt->data[0],
295  BLE_DFU_RESP_VAL_OPER_FAILED);
296  }
297 
298  ble_dfu_evt.evt.ble_dfu_pkt_write.len = 1;
299  ble_dfu_evt.evt.ble_dfu_pkt_write.p_data = &(p_ble_write_evt->data[1]);
300 
301  p_dfu->evt_handler(p_dfu, &ble_dfu_evt);
302  break;
303 
304  case OP_CODE_RECEIVE_INIT:
305  ble_dfu_evt.ble_dfu_evt_type = BLE_DFU_RECEIVE_INIT_DATA;
306 
307  if (p_ble_write_evt->len < PKT_INIT_DFU_PARAM_LEN)
308  {
309  return ble_dfu_response_send(p_dfu,
310  (ble_dfu_procedure_t) p_ble_write_evt->data[0],
311  BLE_DFU_RESP_VAL_OPER_FAILED);
312  }
313 
314  ble_dfu_evt.evt.ble_dfu_pkt_write.len = 1;
315  ble_dfu_evt.evt.ble_dfu_pkt_write.p_data = &(p_ble_write_evt->data[1]);
316 
317  p_dfu->evt_handler(p_dfu, &ble_dfu_evt);
318  break;
319 
320  case OP_CODE_RECEIVE_FW:
321  ble_dfu_evt.ble_dfu_evt_type = BLE_DFU_RECEIVE_APP_DATA;
322 
323  p_dfu->evt_handler(p_dfu, &ble_dfu_evt);
324  break;
325 
326  case OP_CODE_VALIDATE:
327  ble_dfu_evt.ble_dfu_evt_type = BLE_DFU_VALIDATE;
328 
329  p_dfu->evt_handler(p_dfu, &ble_dfu_evt);
330  break;
331 
332  case OP_CODE_ACTIVATE_N_RESET:
333  ble_dfu_evt.ble_dfu_evt_type = BLE_DFU_ACTIVATE_N_RESET;
334 
335  p_dfu->evt_handler(p_dfu, &ble_dfu_evt);
336  break;
337 
338  case OP_CODE_SYS_RESET:
339  ble_dfu_evt.ble_dfu_evt_type = BLE_DFU_SYS_RESET;
340 
341  p_dfu->evt_handler(p_dfu, &ble_dfu_evt);
342  break;
343 
344  case OP_CODE_PKT_RCPT_NOTIF_REQ:
345  if (p_ble_write_evt->len < PKT_RCPT_NOTIF_REQ_LEN)
346  {
347  return (ble_dfu_response_send(p_dfu,
348  BLE_DFU_PKT_RCPT_REQ_PROCEDURE,
349  BLE_DFU_RESP_VAL_NOT_SUPPORTED));
350  }
351 
352  ble_dfu_evt.evt.pkt_rcpt_notif_req.num_of_pkts =
353  uint16_decode(&(p_ble_write_evt->data[1]));
354 
355  if (ble_dfu_evt.evt.pkt_rcpt_notif_req.num_of_pkts == 0)
356  {
357  ble_dfu_evt.ble_dfu_evt_type = BLE_DFU_PKT_RCPT_NOTIF_DISABLED;
358  }
359  else
360  {
361  ble_dfu_evt.ble_dfu_evt_type = BLE_DFU_PKT_RCPT_NOTIF_ENABLED;
362  }
363 
364  p_dfu->evt_handler(p_dfu, &ble_dfu_evt);
365 
366  break;
367 
368  case OP_CODE_IMAGE_SIZE_REQ:
369  ble_dfu_evt.ble_dfu_evt_type = BLE_DFU_BYTES_RECEIVED_SEND;
370 
371  p_dfu->evt_handler(p_dfu, &ble_dfu_evt);
372  break;
373 
374  default:
375  // Unsupported op code.
376  return ble_dfu_response_send(p_dfu,
377  (ble_dfu_procedure_t) p_ble_write_evt->data[0],
378  BLE_DFU_RESP_VAL_NOT_SUPPORTED);
379  }
380  return NRF_SUCCESS;
381 }

447 uint32_t ble_dfu_init(ble_dfu_t * p_dfu, ble_dfu_init_t * p_dfu_init)
448 {
449  if ((p_dfu == NULL) || (p_dfu_init == NULL) || (p_dfu_init->evt_handler == NULL))
450  {
451  return NRF_ERROR_NULL;
452  }
453 
454  p_dfu->conn_handle = BLE_CONN_HANDLE_INVALID;
455 
456  ble_uuid_t service_uuid;
457  uint32_t err_code;
458 
459  const ble_uuid128_t base_uuid128 =
460  {
461  {
462  0x23, 0xD1, 0xBC, 0xEA, 0x5F, 0x78, 0x23, 0x15,
463  0xDE, 0xEF, 0x12, 0x12, 0x00, 0x00, 0x00, 0x00
464  }
465  };
466 
467  service_uuid.uuid = BLE_DFU_SERVICE_UUID;
468 
469  err_code = sd_ble_uuid_vs_add(&base_uuid128, &(service_uuid.type));
470  if (err_code != NRF_SUCCESS)
471  {
472  return err_code;
473  }
474 
475  err_code = sd_ble_gatts_service_add(BLE_GATTS_SRVC_TYPE_PRIMARY,
476  &service_uuid,
477  &(p_dfu->service_handle));
478  if (err_code != NRF_SUCCESS)
479  {
480  return err_code;
481  }
482 
483  p_dfu->uuid_type = service_uuid.type;
484 
485  err_code = dfu_pkt_char_add(p_dfu);
486  if (err_code != NRF_SUCCESS)
487  {
488  return err_code;
489  }
490 
491  err_code = dfu_ctrl_pt_add(p_dfu);
492  if (err_code != NRF_SUCCESS)
493  {
494  return err_code;
495  }
496 
497  err_code = dfu_rev_char_add(p_dfu, p_dfu_init);
498  if (err_code != NRF_SUCCESS)
499  {
500  return err_code;
501  }
502 
503  p_dfu->evt_handler = p_dfu_init->evt_handler;
504 
505  if (p_dfu_init->error_handler != NULL)
506  {
507  p_dfu->error_handler = p_dfu_init->error_handler;
508  }
509 
510  m_is_dfu_service_initialized = true;
511 
512  return NRF_SUCCESS;
513 }

516 void ble_dfu_on_ble_evt(ble_dfu_t * p_dfu, ble_evt_t * p_ble_evt)
517 {
518  if ((p_dfu == NULL) || (p_ble_evt == NULL))
519  {
520  return;
521  }
522 
523  if (p_dfu->evt_handler != NULL)
524  {
525  switch (p_ble_evt->header.evt_id)
526  {
527  case BLE_GAP_EVT_CONNECTED:
528  on_connect(p_dfu, p_ble_evt);
529  break;
530 
531  case BLE_GATTS_EVT_WRITE:
532  on_write(p_dfu, p_ble_evt);
533  break;
534 
535  case BLE_GAP_EVT_DISCONNECTED:
536  on_disconnect(p_dfu, p_ble_evt);
537  break;
538 
539  case BLE_GATTS_EVT_RW_AUTHORIZE_REQUEST:
540  on_rw_auth_req(p_dfu, p_ble_evt);
541  break;
542 
543  default:
544  // No implementation needed.
545  break;
546  }
547  }

580  hvx_params.type = BLE_GATT_HVX_NOTIFICATION;
581  hvx_params.offset = 0;
582  hvx_params.p_len = &index;
583  hvx_params.p_data = m_notif_buffer;
584

586 }

622  ble_dfu_resp_val_t resp_val)
623 {
624  if (p_dfu == NULL)
625  {
626  return NRF_ERROR_NULL;
627  }
628
