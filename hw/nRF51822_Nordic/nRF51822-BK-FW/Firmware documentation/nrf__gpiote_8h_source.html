57 static __INLINE void nrf_gpiote_task_config(uint32_t channel_number, uint32_t pin_number, nrf_gpiote_polarity_t polarity, nrf_gpiote_outinit_t initial_value)
58 {
59  /* Check if the output desired is high or low */
60  if (initial_value == NRF_GPIOTE_INITIAL_VALUE_LOW)
61  {
62  /* Workaround for the OUTINIT PAN. When nrf_gpiote_task_config() is called a glitch happens
63  on the GPIO if the GPIO in question is already assigned to GPIOTE and the pin is in the
64  correct state in GPIOTE but not in the OUT register. */
65  NRF_GPIO->OUTCLR = (1 << pin_number);
66 
67  /* Configure channel to Pin31, not connected to the pin, and configure as a tasks that will set it to proper level */
68  NRF_GPIOTE->CONFIG[channel_number] = (GPIOTE_CONFIG_MODE_Task << GPIOTE_CONFIG_MODE_Pos) |
69  (31UL << GPIOTE_CONFIG_PSEL_Pos) |
70  (GPIOTE_CONFIG_POLARITY_HiToLo << GPIOTE_CONFIG_POLARITY_Pos);
71  }
72  else
73  {
74  /* Workaround for the OUTINIT PAN. When nrf_gpiote_task_config() is called a glitch happens
75  on the GPIO if the GPIO in question is already assigned to GPIOTE and the pin is in the
76  correct state in GPIOTE but not in the OUT register. */
77  NRF_GPIO->OUTSET = (1 << pin_number);
78 
79  /* Configure channel to Pin31, not connected to the pin, and configure as a tasks that will set it to proper level */
80  NRF_GPIOTE->CONFIG[channel_number] = (GPIOTE_CONFIG_MODE_Task << GPIOTE_CONFIG_MODE_Pos) |
81  (31UL << GPIOTE_CONFIG_PSEL_Pos) |
82  (GPIOTE_CONFIG_POLARITY_LoToHi << GPIOTE_CONFIG_POLARITY_Pos);
83  }
84 
85  /* Three NOPs are required to make sure configuration is written before setting tasks or getting events */
86  __NOP();
87  __NOP();
88  __NOP();
89 
90  /* Launch the task to take the GPIOTE channel output to the desired level */
91  NRF_GPIOTE->TASKS_OUT[channel_number] = 1;
92 
93 
94  /* Finally configure the channel as the caller expects. If OUTINIT works, the channel is configured properly.
95  If it does not, the channel output inheritance sets the proper level. */
96  NRF_GPIOTE->CONFIG[channel_number] = (GPIOTE_CONFIG_MODE_Task << GPIOTE_CONFIG_MODE_Pos) |
97  ((uint32_t)pin_number << GPIOTE_CONFIG_PSEL_Pos) |
98  ((uint32_t)polarity << GPIOTE_CONFIG_POLARITY_Pos) |
99  ((uint32_t)initial_value << GPIOTE_CONFIG_OUTINIT_Pos);
100 
101  /* Three NOPs are required to make sure configuration is written before setting tasks or getting events */
102  __NOP();
103  __NOP();
104  __NOP();
105 }
