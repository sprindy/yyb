48 static void on_write(ble_bcs_t * p_bcs, ble_evt_t * p_ble_evt)
49 {
50  ble_gatts_evt_write_t * p_evt_write = &p_ble_evt->evt.gatts_evt.params.write;
51 
52  if (p_bcs->beacon_write_handler == NULL)
53  {
54  return;
55  }
56 
57  if ((p_evt_write->handle == p_bcs->beacon_maj_min_char_handles.value_handle) &&
58  (p_evt_write->len == BCS_DATA_MAJ_MIN_LEN))
59  {
60  p_bcs->beacon_write_handler(p_bcs, beacon_maj_min_data, p_evt_write->data);
61  }
62 
63  if ((p_evt_write->handle == p_bcs->beacon_calib_char_handles.value_handle) &&
64  (p_evt_write->len == BCS_DATA_CALIB_LEN))
65  {
66  p_bcs->beacon_write_handler(p_bcs, beacon_measured_rssi_data, p_evt_write->data);
67  }
68 
69  if ((p_evt_write->handle == p_bcs->beacon_id_char_handles.value_handle) &&
70  (p_evt_write->len == BCS_DATA_ID_LEN))
71  {
72  p_bcs->beacon_write_handler(p_bcs, beacon_uuid_data, p_evt_write->data);
73  }
74 
75  if ((p_evt_write->handle == p_bcs->beacon_comp_id_char_handles.value_handle) &&
76  (p_evt_write->len == BCS_DATA_COMPANY_ID_LEN))
77  {
78  p_bcs->beacon_write_handler(p_bcs, beacon_company_id_data, p_evt_write->data);
79  }
80 
81  if ((p_evt_write->handle == p_bcs->beacon_adv_int_char_handles.value_handle) &&
82  (p_evt_write->len == BCS_DATA_ADV_INT_LEN))
83  {
84  p_bcs->beacon_write_handler(p_bcs, beacon_adv_interval_data, p_evt_write->data);
85  }
86 
87  if ((p_evt_write->handle == p_bcs->beacon_led_char_handles.value_handle) &&
88  (p_evt_write->len == BCS_DATA_LED_LEN))
89  {
90  p_bcs->beacon_write_handler(p_bcs, beacon_led_data, p_evt_write->data);
91  }
92 }

95 void ble_bcs_on_ble_evt(ble_bcs_t * p_bcs, ble_evt_t * p_ble_evt)
96 {
97  switch (p_ble_evt->header.evt_id)
98  {
99  case BLE_GAP_EVT_CONNECTED:
100  on_connect(p_bcs, p_ble_evt);
101  break;
102 
103  case BLE_GAP_EVT_DISCONNECTED:
104  on_disconnect(p_bcs, p_ble_evt);
105  break;
106 
107  case BLE_GATTS_EVT_WRITE:
108  on_write(p_bcs, p_ble_evt);
109  break;
110 
111  default:
112  break;
113  }
114 }

438 uint32_t ble_bcs_init(ble_bcs_t * p_bcs, const ble_bcs_init_t * p_bcs_init)
439 {
440  uint32_t err_code;
441  ble_uuid_t ble_uuid;
442 
443  // Initialize service structure
444  p_bcs->conn_handle = BLE_CONN_HANDLE_INVALID;
445  p_bcs->beacon_write_handler = p_bcs_init->beacon_write_handler;
446 
447  // Add base UUID to softdevice's internal list.
448  ble_uuid128_t base_uuid = BCS_UUID_BASE;
449  err_code = sd_ble_uuid_vs_add(&base_uuid, &p_bcs->uuid_type);
450  if (err_code != NRF_SUCCESS)
451  {
452  return err_code;
453  }
454 
455  ble_uuid.type = p_bcs->uuid_type;
456  ble_uuid.uuid = BCS_UUID_SERVICE;
457 
458  err_code = sd_ble_gatts_service_add(BLE_GATTS_SRVC_TYPE_PRIMARY, &ble_uuid, &p_bcs->service_handle);
459  if (err_code != NRF_SUCCESS)
460  {
461  return err_code;
462  }
463 
464  err_code = beacon_maj_min_char_add(p_bcs, p_bcs_init);
465  if (err_code != NRF_SUCCESS)
466  {
467  return err_code;
468  }
469 
470  err_code = beacon_calib_char_add(p_bcs, p_bcs_init);
471  if (err_code != NRF_SUCCESS)
472  {
473  return err_code;
474  }
475 
476  err_code = beacon_id_char_add(p_bcs, p_bcs_init);
477  if (err_code != NRF_SUCCESS)
478  {
479  return err_code;
480  }
481 
482  err_code = beacon_company_id_char_add(p_bcs, p_bcs_init);
483  if (err_code != NRF_SUCCESS)
484  {
485  return err_code;
486  }
487 
488  err_code = beacon_adv_interval_char_add(p_bcs, p_bcs_init);
489  if (err_code != NRF_SUCCESS)
490  {
491  return err_code;
492  }
493 
494  err_code = beacon_led_char_add(p_bcs, p_bcs_init);
495  if (err_code != NRF_SUCCESS)

497  return err_code;
498  }
499 
500  return NRF_SUCCESS;
