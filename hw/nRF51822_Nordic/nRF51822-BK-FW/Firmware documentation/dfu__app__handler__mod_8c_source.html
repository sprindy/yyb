41 static void interrupts_disable(void)
42 {
43  uint32_t interrupt_setting_mask;
44  uint32_t irq = 0; // We start from first interrupt, i.e. interrupt 0.
45 
46  // Fetch the current interrupt settings.
47  interrupt_setting_mask = NVIC->ISER[0];
48 
49  for (; irq < MAX_NUMBER_INTERRUPTS; irq++)
50  {
51  if (interrupt_setting_mask & (IRQ_ENABLED << irq))
52  {
53  // The interrupt was enabled, and hence disable it.
54  NVIC_DisableIRQ((IRQn_Type)irq);
55  }
56  }
57 }

81 void dfu_app_on_dfu_evt(ble_dfu_t * p_dfu, ble_dfu_evt_t * p_evt)
82 {
83  switch (p_evt->ble_dfu_evt_type)
84  {
85  case BLE_DFU_START:
86  // Starting the bootloader - will cause reset.
87  bootloader_start();
88  break;
89 
90  case BLE_DFU_VALIDATE:
91  case BLE_DFU_ACTIVATE_N_RESET:
92  case BLE_DFU_SYS_RESET:
93  case BLE_DFU_RECEIVE_INIT_DATA:
94  case BLE_DFU_RECEIVE_APP_DATA:
95  case BLE_DFU_PACKET_WRITE:
96  case BLE_DFU_PKT_RCPT_NOTIF_ENABLED:
97  case BLE_DFU_PKT_RCPT_NOTIF_DISABLED:
98  case BLE_DFU_BYTES_RECEIVED_SEND:
99  default:
100  {
101  // Unsupported event received from DFU Service.
102  // Send back BLE_DFU_RESP_VAL_NOT_SUPPORTED message to peer.
103  uint32_t err_code = ble_dfu_response_send(p_dfu,
104  BLE_DFU_START_PROCEDURE,
105  BLE_DFU_RESP_VAL_NOT_SUPPORTED);
106  APP_ERROR_CHECK(err_code);
107  }
108  break;
109  }
