82 uint32_t app_sched_event_put(void * p_event_data,
83  uint16_t event_data_size,
84  app_sched_event_handler_t handler)
85 {
86  uint32_t err_code;
87 
88  if (event_data_size <= m_queue_event_size)
89  {
90  uint16_t event_index = 0xFFFF;
91 
92  CRITICAL_REGION_ENTER();
93 
94  if (!APP_SCHED_QUEUE_FULL())
95  {
96  event_index = m_queue_end_index;
97  m_queue_end_index = next_index(m_queue_end_index);
98  }
99 
100  CRITICAL_REGION_EXIT();
101 
102  if (event_index != 0xFFFF)
103  {
104  //NOTE: This can be done outside the critical region since the event consumer will
105  //always be called from the main loop, and will thus never interrupt this code.
106  m_queue_event_headers[event_index].handler = handler;
107 
108  if ((p_event_data != NULL) && (event_data_size > 0))
109  {
110  memcpy(&m_queue_event_data[event_index * m_queue_event_size],
111  p_event_data,
112  event_data_size);
113  m_queue_event_headers[event_index].event_data_size = event_data_size;
114  }
115  else
116  {
117  m_queue_event_headers[event_index].event_data_size = 0;
118  }
119 
120  err_code = NRF_SUCCESS;
121  }
122  else
123  {
124  err_code = NRF_ERROR_NO_MEM;
125  }
126  }
127  else
128  {
129  err_code = NRF_ERROR_INVALID_LENGTH;
130  }
131 
132  return err_code;
133 }

144 static uint32_t app_sched_event_get(void * * pp_event_data,
145  uint16_t * p_event_data_size,
146  app_sched_event_handler_t * p_event_handler)
147 {
148  uint32_t err_code = NRF_ERROR_NOT_FOUND;
149 
150  if (!APP_SCHED_QUEUE_EMPTY())
151  {
152  uint16_t event_index;
153 
154  //NOTE: There is no need for a critical region here, as this function will only be called
155  //from app_sched_execute() from inside the main loop, so it will never interrupt
156  //app_sched_event_put(). Also, updating of (i.e. writing to) the start index will be
157  //an atomic operation.
158  event_index = m_queue_start_index;
159  m_queue_start_index = next_index(m_queue_start_index);
160 
161  *pp_event_data = &m_queue_event_data[event_index * m_queue_event_size];
162  *p_event_data_size = m_queue_event_headers[event_index].event_data_size;
163  *p_event_handler = m_queue_event_headers[event_index].handler;
164 
165  err_code = NRF_SUCCESS;
166  }
167 
168  return err_code;
169 }
