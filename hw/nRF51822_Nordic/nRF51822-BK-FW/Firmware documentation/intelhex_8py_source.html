93  def _decode_record(self, s, line=0):
94  '''Decode one record of HEX file.
95 
96  @param s line with HEX record.
97  @param line line number (for error messages).
98 
99  @raise EndOfFile if EOF record encountered.
100  '''
101  s = s.rstrip('\r\n')
102  if not s:
103  return # empty line
104 
105  if s[0] == ':':
106  try:
107  bin = array('B', unhexlify(asbytes(s[1:])))
108  except (TypeError, ValueError):
109  # this might be raised by unhexlify when odd hexascii digits
110  raise HexRecordError(line=line)
111  length = len(bin)
112  if length < 5:
113  raise HexRecordError(line=line)
114  else:
115  raise HexRecordError(line=line)
116 
117  record_length = bin[0]
118  if length != (5 + record_length):
119  raise RecordLengthError(line=line)
120 
121  addr = bin[1]*256 + bin[2]
122 
123  record_type = bin[3]
124  if not (0 <= record_type <= 5):
125  raise RecordTypeError(line=line)
126 
127  crc = sum(bin)
128  crc &= 0x0FF
129  if crc != 0:
130  raise RecordChecksumError(line=line)
131 
132  if record_type == 0:
133  # data record
134  addr += self._offset
135  for i in xrange(4, 4+record_length):
136  if not self._buf.get(addr, None) is None:
137  raise AddressOverlapError(address=addr, line=line)
138  self._buf[addr] = bin[i]
139  addr += 1 # FIXME: addr should be wrapped
140  # BUT after 02 record (at 64K boundary)
141  # and after 04 record (at 4G boundary)
142 
143  elif record_type == 1:
144  # end of file record
145  if record_length != 0:
146  raise EOFRecordError(line=line)
147  raise _EndOfFile
148 
149  elif record_type == 2:
150  # Extended 8086 Segment Record
151  if record_length != 2 or addr != 0:
152  raise ExtendedSegmentAddressRecordError(line=line)
153  self._offset = (bin[4]*256 + bin[5]) * 16
154 
155  elif record_type == 4:
156  # Extended Linear Address Record
157  if record_length != 2 or addr != 0:
158  raise ExtendedLinearAddressRecordError(line=line)
159  self._offset = (bin[4]*256 + bin[5]) * 65536
160 
161  elif record_type == 3:
162  # Start Segment Address Record
163  if record_length != 4 or addr != 0:
164  raise StartSegmentAddressRecordError(line=line)
165  if self.start_addr:
166  raise DuplicateStartAddressRecordError(line=line)
167  self.start_addr = {'CS': bin[4]*256 + bin[5],
168  'IP': bin[6]*256 + bin[7],
169  }
170 
171  elif record_type == 5:
172  # Start Linear Address Record
173  if record_length != 4 or addr != 0:
174  raise StartLinearAddressRecordError(line=line)
175  if self.start_addr:
176  raise DuplicateStartAddressRecordError(line=line)
177  self.start_addr = {'EIP': (bin[4]*16777216 +
178  bin[5]*65536 +
179  bin[6]*256 +
180  bin[7]),
181  }

281  def _get_start_end(self, start=None, end=None, size=None):

284  invoke this method with both start and end as None.
285  """

445  def __getitem__(self, addr):
446  ''' Get requested byte from address.
447  @param addr address of byte.
448  @return byte if address exists in HEX file, or self.padding

472  def __setitem__(self, addr, byte):
473  """Set byte at address."""
474  t = type(addr)
475  if t in (int, long):
476  if addr < 0:
477  raise TypeError('Address should be >= 0.')

493  start = stop - len(byte)
494  elif stop is None:

497  raise TypeError('start address cannot be negative')

499  raise TypeError('stop address cannot be negative')

534 
535  @param f filename or file-like object for writing
536  @param write_start_addr enable or disable writing start address
537  record to file (enabled by default).
538  If there is no start address in obj, nothing
539  will be written regardless of this setting.
540  """
541  fwrite = getattr(f, "write", None)
542  if fwrite:
543  fobj = f
544  fclose = None
545  else:
546  fobj = open(f, 'w')
547  fwrite = fobj.write
548  fclose = fobj.close

550  # Translation table for uppercasing hex ascii string.
551  # timeit shows that using hexstr.translate(table)
552  # is faster than hexstr.upper():
553  # 0.452ms vs. 0.652ms (translate vs. upper)
554  if sys.version_info[0] >= 3:
555  table = bytes(range(256)).upper()
556  else:
557  table = ''.join(chr(i).upper() for i in range(256))
558 
559 
560 
561  # start address record if any
562  if self.start_addr and write_start_addr:
563  keys = self.start_addr.keys()
564  keys.sort()
565  bin = array('B', asbytes('\0'*9))
566  if keys == ['CS','IP']:
567  # Start Segment Address Record
568  bin[0] = 4 # reclen
569  bin[1] = 0 # offset msb
570  bin[2] = 0 # offset lsb
571  bin[3] = 3 # rectyp
572  cs = self.start_addr['CS']
573  bin[4] = (cs >> 8) & 0x0FF
574  bin[5] = cs & 0x0FF
575  ip = self.start_addr['IP']
576  bin[6] = (ip >> 8) & 0x0FF
577  bin[7] = ip & 0x0FF
578  bin[8] = (-sum(bin)) & 0x0FF # chksum
579  fwrite(':' +
580  asstr(hexlify(bin.tostring()).translate(table)) +
581  '\n')
582  elif keys == ['EIP']:
583  # Start Linear Address Record
584  bin[0] = 4 # reclen
585  bin[1] = 0 # offset msb
586  bin[2] = 0 # offset lsb
587  bin[3] = 5 # rectyp
588  eip = self.start_addr['EIP']
589  bin[4] = (eip >> 24) & 0x0FF
590  bin[5] = (eip >> 16) & 0x0FF
591  bin[6] = (eip >> 8) & 0x0FF
592  bin[7] = eip & 0x0FF
593  bin[8] = (-sum(bin)) & 0x0FF # chksum
594  fwrite(':' +
595  asstr(hexlify(bin.tostring()).translate(table)) +
596  '\n')
597  else:
598  if fclose:
599  fclose()
600  raise InvalidStartAddressValueError(start_addr=self.start_addr)
601 
602  # data
603  addresses = self._buf.keys()
604  addresses.sort()
605  addr_len = len(addresses)
606  if addr_len:
607  minaddr = addresses[0]
608  maxaddr = addresses[-1]
609 
610  if maxaddr > 65535:
611  need_offset_record = True
612  else:
613  need_offset_record = False
614  high_ofs = 0
615 
616  cur_addr = minaddr
617  cur_ix = 0
618 
619  while cur_addr <= maxaddr:
620  if need_offset_record:
621  bin = array('B', asbytes('\0'*7))
622  bin[0] = 2 # reclen
623  bin[1] = 0 # offset msb
624  bin[2] = 0 # offset lsb
625  bin[3] = 4 # rectyp
626  high_ofs = int(cur_addr>>16)
627  b = divmod(high_ofs, 256)
628  bin[4] = b[0] # msb of high_ofs
629  bin[5] = b[1] # lsb of high_ofs
630  bin[6] = (-sum(bin)) & 0x0FF # chksum
631  fwrite(':' +
632  asstr(hexlify(bin.tostring()).translate(table)) +
633  '\n')
634 
635  while True:
636  # produce one record
637  low_addr = cur_addr & 0x0FFFF
638  # chain_len off by 1
639  chain_len = min(15, 65535-low_addr, maxaddr-cur_addr)
640 
641  # search continuous chain
642  stop_addr = cur_addr + chain_len
643  if chain_len:
644  ix = bisect_right(addresses, stop_addr,
645  cur_ix,
646  min(cur_ix+chain_len+1, addr_len))
647  chain_len = ix - cur_ix # real chain_len
648  # there could be small holes in the chain
649  # but we will catch them by try-except later
650  # so for big continuous files we will work
651  # at maximum possible speed
652  else:
653  chain_len = 1 # real chain_len
654 
655  bin = array('B', asbytes('\0'*(5+chain_len)))
656  b = divmod(low_addr, 256)
657  bin[1] = b[0] # msb of low_addr
658  bin[2] = b[1] # lsb of low_addr
659  bin[3] = 0 # rectype
660  try: # if there is small holes we'll catch them
661  for i in range(chain_len):
662  bin[4+i] = self._buf[cur_addr+i]
663  except KeyError:
664  # we catch a hole so we should shrink the chain
665  chain_len = i
666  bin = bin[:5+i]
667  bin[0] = chain_len
668  bin[4+chain_len] = (-sum(bin)) & 0x0FF # chksum
669  fwrite(':' +
670  asstr(hexlify(bin.tostring()).translate(table)) +
671  '\n')
672 
673  # adjust cur_addr/cur_ix
674  cur_ix += chain_len
675  if cur_ix < addr_len:
676  cur_addr = addresses[cur_ix]
677  else:
678  cur_addr = maxaddr + 1
679  break
680  high_addr = int(cur_addr>>16)
681  if high_addr > high_ofs:
682  break
683 
684  # end-of-file record
685  fwrite(":00000001FF\n")
686  if fclose:
687  fclose()

792  def merge(self, other, overlap='error'):
793  """Merge content of other IntelHex object into current object (self).
794  @param other other IntelHex object.
795  @param overlap action on overlap of data or starting addr:
796  - error: raising OverlapError;
797  - ignore: ignore other data and keep current data
798  in overlapping region;
799  - replace: replace data with other data
800  in overlapping region.
801 
802  @raise TypeError if other is not instance of IntelHex
803  @raise ValueError if other is the same object as self
804  (it can't merge itself)
805  @raise ValueError if overlap argument has incorrect value
806  @raise AddressOverlapError on overlapped data
807  """
808  # check args
809  if not isinstance(other, IntelHex):
810  raise TypeError('other should be IntelHex object')
811  if other is self:
812  raise ValueError("Can't merge itself")
813  if overlap not in ('error', 'ignore', 'replace'):
814  raise ValueError("overlap argument should be either "
815  "'error', 'ignore' or 'replace'")
816  # merge data
817  this_buf = self._buf
818  other_buf = other._buf
819  for i in other_buf:
820  if i in this_buf:
821  if overlap == 'error':
822  raise AddressOverlapError(
823  'Data overlapped at address 0x%X' % i)
824  elif overlap == 'ignore':
825  continue
826  this_buf[i] = other_buf[i]
827  # merge start_addr
828  if self.start_addr != other.start_addr:
829  if self.start_addr is None: # set start addr from other
830  self.start_addr = other.start_addr
831  elif other.start_addr is None: # keep existing start addr
832  pass
833  else: # conflict
834  if overlap == 'error':
835  raise AddressOverlapError(
836  'Starting addresses are different')
837  elif overlap == 'replace':
838  self.start_addr = other.start_addr
