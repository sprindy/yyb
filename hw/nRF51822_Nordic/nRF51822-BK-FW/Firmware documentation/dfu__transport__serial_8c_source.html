134 static uint32_t data_queue_element_alloc(uint8_t * p_element_index, uint8_t packet_type)
135 {
136  uint32_t retval;
137  uint32_t index;
138 
139  retval = NRF_ERROR_NO_MEM;
140 
141  if (INVALID_PACKET == packet_type)
142  {
143  retval = NRF_ERROR_INVALID_PARAM;
144  }
145  else if (true == DATA_QUEUE_FULL())
146  {
147  retval = NRF_ERROR_NO_MEM;
148  }
149  else
150  {
151  for (index = 0; index < MAX_BUFFERS; index++)
152  {
153  if (INVALID_PACKET == DATA_QUEUE_ELEMENT_GET_PTYPE(index))
154  {
155  // Found a free element: allocate, and end search.
156  *p_element_index = index;
157  DATA_QUEUE_ELEMENT_SET_PTYPE(index, packet_type);
158  retval = NRF_SUCCESS;
159  m_data_queue.count++;
160  break;
161  }
162  }
163  }
164 
165  return retval;
166 }

194 static void process_dfu_packet(void * p_event_data, uint16_t event_size)
195 {
196  uint32_t retval;
197  uint32_t index;
198  dfu_update_packet_t * packet;
199 
200  while (false == DATA_QUEUE_EMPTY())
201  {
202  // Fetch the element to be processed.
203  for (index = 0; index < MAX_BUFFERS ; index++)
204  {
205  packet = &m_data_queue.data_packet[index];
206  if (INVALID_PACKET != packet->packet_type)
207  {
208 
209  switch (DATA_QUEUE_ELEMENT_GET_PTYPE(index))
210  {
211  case DATA_PACKET:
212  (void)dfu_data_pkt_handle(packet);
213  break;
214 
215  case START_PACKET:
216  packet->params.start_packet =
217  (dfu_start_packet_t*)packet->params.data_packet.p_data_packet;
218  retval = dfu_start_pkt_handle(packet);
219  APP_ERROR_CHECK(retval);
220  break;
221 
222  case INIT_PACKET:
223  (void)dfu_init_pkt_handle(packet);
224  retval = dfu_init_pkt_complete();
225  APP_ERROR_CHECK(retval);
226  break;
227 
228  case STOP_DATA_PACKET:
229  (void)dfu_image_validate();
230  (void)dfu_image_activate();
231 
232  // Break the loop by returning.
233  return;
234 
235  default:
236  // No implementation needed.
237  break;
238  }
239 
240  // Free the processed element.
241  retval = data_queue_element_free(index);
242  APP_ERROR_CHECK(retval);
243  }
244  }
245  }

249 void rpc_transport_event_handler(hci_transport_evt_t event)
250 {
251  uint32_t retval;
252  uint16_t rpc_cmd_length_read;
253  uint8_t * p_rpc_cmd_buffer;
254  uint8_t element_index;
255 
256  retval = hci_transport_rx_pkt_extract(&p_rpc_cmd_buffer, &rpc_cmd_length_read);
257  if (NRF_SUCCESS == retval)
258  {
259  // Verify if the data queue can buffer the packet.
260  retval = data_queue_element_alloc(&element_index, p_rpc_cmd_buffer[0]);
261  if (NRF_SUCCESS == retval)
262  {
263  //subtract 1 since we are interested in payload length and not the type field.
264  DATA_QUEUE_ELEMENT_SET_PLEN(element_index,(rpc_cmd_length_read / sizeof(uint32_t)) - 1);
265  DATA_QUEUE_ELEMENT_COPY_PDATA(element_index, &p_rpc_cmd_buffer[4]);
266  retval = app_sched_event_put(NULL, 0, process_dfu_packet);
267  }
268  }
269 
270  if (NRF_SUCCESS != retval)
271  {
272  // Free the packet that could not be processed.
273  retval = hci_transport_rx_pkt_consume(p_rpc_cmd_buffer);
274  APP_ERROR_CHECK(retval);
275  }
276 }
