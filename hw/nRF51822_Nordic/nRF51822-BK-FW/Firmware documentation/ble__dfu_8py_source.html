59 def calc_crc16(binfile):
60  crc = 0xFFFF
61 
62  for b in binfile:
63  crc = (crc >> 8 & 0x00FF) | (crc << 8 & 0xFF00)
64  crc = crc ^ b
65  crc = crc ^ ( ( crc & 0x00FF) >> 4 )
66  crc = crc ^ ( ( crc << 8) << 4)
67  crc = crc ^ ( ( crc & 0x00FF) << 4) << 1
68  return crc & 0xFFFF
