192 uint32_t pstorage_register(pstorage_module_param_t * p_module_param,
193  pstorage_handle_t * p_block_id)
194 {
195 
196 
197  VERIFY_MODULE_INITIALIZED();
198  NULL_PARAM_CHECK(p_module_param);
199  NULL_PARAM_CHECK(p_block_id);
200  NULL_PARAM_CHECK(p_module_param->cb);
201  BLOCK_SIZE_CHECK(p_module_param->block_size);
202  BLOCK_COUNT_CHECK(p_module_param->block_count, p_module_param->block_size);
203 
204 #ifndef SUPPORT_MODULES_LARGER_THAN_PAGE
205  MODULE_SIZE_CHECK(((uint16_t)(p_module_param->block_size)*(p_module_param->block_count)));
206 #endif
207 
208 
209  if (m_next_app_instance == PSTORAGE_MAX_APPLICATIONS)
210  {
211  return NRF_ERROR_NO_MEM;
212  }
213 
214  p_block_id->module_id = m_next_app_instance;
215  p_block_id->block_id = m_next_page_addr;
216 
217  m_app_table[m_next_app_instance].base_id = p_block_id->block_id;
218  m_app_table[m_next_app_instance].cb = p_module_param->cb;
219  m_app_table[m_next_app_instance].block_size = p_module_param->block_size;
220  m_app_table[m_next_app_instance].block_count = p_module_param->block_count;
221 
222  // Calculate number of flash pages allocated for the device.
223 
224 
225 #ifdef SUPPORT_MODULES_LARGER_THAN_PAGE
226  uint16_t page_count = 0;
227  uint32_t total_size;
228  total_size = p_module_param->block_size * p_module_param->block_count;
229  do
230  {
231  page_count++;
232 
233  if (total_size > PSTORAGE_FLASH_PAGE_SIZE)
234  {
235  total_size -= PSTORAGE_FLASH_PAGE_SIZE;
236  }
237  else
238  {
239  total_size = 0;
240  }
241 
242  m_next_page_addr += PSTORAGE_FLASH_PAGE_SIZE;
243 
244  } while (total_size >= PSTORAGE_FLASH_PAGE_SIZE);
245  m_app_table[m_next_app_instance].no_of_pages = page_count;
246 #else
247  m_app_table[m_next_app_instance].no_of_pages = 1;
248  m_next_page_addr += PSTORAGE_FLASH_PAGE_SIZE;
249 #endif
250  m_next_app_instance++;
251 
252  return NRF_SUCCESS;
253 }

327 uint32_t pstorage_update(pstorage_handle_t * p_dest,
328  uint8_t * p_src,
329  pstorage_size_t size,
330  pstorage_size_t offset)
331 {
332 
333  VERIFY_MODULE_INITIALIZED();
334  NULL_PARAM_CHECK(p_src);
335  NULL_PARAM_CHECK(p_dest);
336  MODULE_ID_RANGE_CHECK(p_dest);
337  BLOCK_ID_RANGE_CHECK(p_dest);
338  SIZE_CHECK(p_dest, size);
339  OFFSET_CHECK(p_dest, offset, size);
340 
341  uint32_t *p_swap_addr = (uint32_t *)PSTORAGE_SWAP_ADDR;
342  uint32_t *p_page_addr1 = (uint32_t *) PAGE_BASE_ADDR(p_dest->block_id + offset);
343 #ifdef SUPPORT_MODULES_LARGER_THAN_PAGE
344  uint32_t *p_page_addr2 = (uint32_t *) PAGE_BASE_ADDR(p_dest->block_id + offset + size-1);
345 #endif
346  // block may not be aligned with page, so head must include all trailing blocks
347  uint16_t head_word_count = (uint32_t)((uint32_t *)p_dest->block_id - p_page_addr1) + (offset)/sizeof(uint32_t);
348  uint16_t body_word_count = size/sizeof(uint32_t);
349  uint16_t tail_word_count;
350  uint32_t retval;
351 
352  // Verify word alignment.
353  if ((!is_word_aligned(p_src)) || (!is_word_aligned((void *)(uint32_t)offset)))
354  {
355  return NRF_ERROR_INVALID_ADDR;
356  }
357  // erase swap page
358  (void) ble_flash_page_erase(PSTORAGE_SWAP_ADDR / BLE_FLASH_PAGE_SIZE);
359 #ifdef SUPPORT_MODULES_LARGER_THAN_PAGE
360  if (p_page_addr1 == p_page_addr2)
361  {
362 #endif
363  // tail is in the same page as head
364  tail_word_count = BLE_FLASH_PAGE_SIZE/sizeof(uint32_t) - head_word_count - body_word_count;
365  // copy of the head
366  if (head_word_count)
367  {
368  retval = ble_flash_block_write(p_swap_addr, p_page_addr1, head_word_count );
369  if (retval != NRF_SUCCESS)
370  {
371  app_notify(p_dest, p_src, PSTORAGE_UPDATE_OP_CODE, size, retval);
372  return retval;
373  }
374  }
375  // copy of the body
376  retval = ble_flash_block_write(p_swap_addr+head_word_count, (uint32_t *)p_src, body_word_count);
377  if (retval != NRF_SUCCESS)
378  {
379  app_notify(p_dest, p_src, PSTORAGE_UPDATE_OP_CODE, size, retval);
380  return retval;
381  }
382  // copy of the tail
383  if (tail_word_count)
384  {
385  retval = ble_flash_block_write(p_swap_addr+head_word_count+body_word_count, p_page_addr1+head_word_count+body_word_count, tail_word_count );
386  if (retval != NRF_SUCCESS)
387  {
388  app_notify(p_dest, p_src, PSTORAGE_UPDATE_OP_CODE, size, retval);
389  return retval;
390  }
391  }
392  // erase active page
393  (void) ble_flash_page_erase((uint32_t)p_page_addr1 / BLE_FLASH_PAGE_SIZE);
394  // restore updated page
395  retval = ble_flash_block_write(p_page_addr1, p_swap_addr, BLE_FLASH_PAGE_SIZE/sizeof(uint32_t));
396 #ifdef SUPPORT_MODULES_LARGER_THAN_PAGE
397  }
398  else
399  {
400  // tail is in NOT in the same page as head - need to swap twice
401  uint16_t body1_word_count = BLE_FLASH_PAGE_SIZE/sizeof(uint32_t) - head_word_count;
402  uint16_t body2_word_count = body_word_count - body1_word_count;
403  tail_word_count = BLE_FLASH_PAGE_SIZE/sizeof(uint32_t) - body2_word_count;
404  // copy head
405  retval = ble_flash_block_write(p_swap_addr, p_page_addr1, head_word_count );
406  if (retval != NRF_SUCCESS)
407  {
408  app_notify(p_dest, p_src, PSTORAGE_UPDATE_OP_CODE, size, retval);
409  return retval;
410  }
411  // copy body1
412  retval = ble_flash_block_write(p_swap_addr, (uint32_t *)p_src, body1_word_count );
413  if (retval != NRF_SUCCESS)
414  {
415  app_notify(p_dest, p_src, PSTORAGE_UPDATE_OP_CODE, size, retval);
416  return retval;
417  }
418  // erase active page
419  (void) ble_flash_page_erase((uint32_t)p_page_addr1 / BLE_FLASH_PAGE_SIZE);
420  // restore updated page1
421  retval = ble_flash_block_write(p_page_addr1, p_swap_addr, BLE_FLASH_PAGE_SIZE/sizeof(uint32_t));
422  if (retval != NRF_SUCCESS)
423  {
424  app_notify(p_dest, p_src, PSTORAGE_UPDATE_OP_CODE, size, retval);
425  return retval;
426  }
427  // erase swap page
428  (void) ble_flash_page_erase(PSTORAGE_SWAP_ADDR / BLE_FLASH_PAGE_SIZE);
429  // copy body2
430  retval = ble_flash_block_write(p_swap_addr, (uint32_t *)p_src + body1_word_count, body2_word_count );
431  if (retval != NRF_SUCCESS)
432  {
433  app_notify(p_dest, p_src, PSTORAGE_UPDATE_OP_CODE, size, retval);
434  return retval;
435  }
436  // copy tail
437  retval = ble_flash_block_write(p_swap_addr, p_page_addr2+body2_word_count, tail_word_count );
438  if (retval != NRF_SUCCESS)

450  return retval;
451  }
452  }

486 uint32_t pstorage_clear(pstorage_handle_t * p_dest, pstorage_size_t size)
487 {
488  uint32_t page_addr;
489  uint32_t retval;
490  uint16_t page_count;
491 
492  VERIFY_MODULE_INITIALIZED();

494  MODULE_ID_RANGE_CHECK(p_dest);
495 
496  page_addr = p_dest->block_id / BLE_FLASH_PAGE_SIZE;
497 
498  retval = NRF_SUCCESS;
499 
500  for (page_count = 0; page_count < m_app_table[p_dest->module_id].no_of_pages; page_count++)
501  {
502  retval = ble_flash_page_erase(page_addr);
503  page_addr++;
504  if (retval != NRF_SUCCESS)
505  {
506  break;
507  }
508  }
509  app_notify(p_dest, NULL, PSTORAGE_CLEAR_OP_CODE, size, retval);
510  return retval;
