221 static void timer_list_insert(app_timer_id_t timer_id)
222 {
223  timer_node_t * p_timer = &mp_nodes[timer_id];
224 
225  if (m_timer_id_head == TIMER_NULL)
226  {
227  m_timer_id_head = timer_id;
228  }
229  else
230  {
231  if (p_timer->ticks_to_expire <= mp_nodes[m_timer_id_head].ticks_to_expire)
232  {
233  mp_nodes[m_timer_id_head].ticks_to_expire -= p_timer->ticks_to_expire;
234 
235  p_timer->next = m_timer_id_head;
236  m_timer_id_head = timer_id;
237  }
238  else
239  {
240  app_timer_id_t previous;
241  app_timer_id_t current;
242  uint32_t ticks_to_expire;
243 
244  ticks_to_expire = p_timer->ticks_to_expire;
245  previous = m_timer_id_head;
246  current = m_timer_id_head;
247 
248  while ((current != TIMER_NULL) && (ticks_to_expire > mp_nodes[current].ticks_to_expire))
249  {
250  ticks_to_expire -= mp_nodes[current].ticks_to_expire;
251  previous = current;
252  current = mp_nodes[current].next;
253  }
254 
255  if (current != TIMER_NULL)
256  {
257  mp_nodes[current].ticks_to_expire -= ticks_to_expire;
258  }
259 
260  p_timer->ticks_to_expire = ticks_to_expire;
261  p_timer->next = current;
262  mp_nodes[previous].next = timer_id;
263  }
264  }

272 static void timer_list_remove(app_timer_id_t timer_id)
273 {
274  app_timer_id_t previous;
275  app_timer_id_t current;
276  uint32_t timeout;
277 
278  // Find the timer's position in timer list.
279  previous = m_timer_id_head;
280  current = previous;
281 
282  while (current != TIMER_NULL)
283  {
284  if (current == timer_id)
285  {
286  break;
287  }
288  previous = current;
289  current = mp_nodes[current].next;
290  }
291 
292  // Timer not in active list.
293  if (current == TIMER_NULL)
294  {
295  return;
296  }
297 
298  // Timer is the first in the list.
299  if (previous == current)
300  {
301  m_timer_id_head = mp_nodes[m_timer_id_head].next;
302  }
303 
304  // Remaining timeout between next timeout.
305  timeout = mp_nodes[current].ticks_to_expire;
306 
307  // Link previous timer with next of this timer, i.e. removing the timer from list.
308  mp_nodes[previous].next = mp_nodes[current].next;
309 
310  // If this is not the last timer, increment the next timer by this timer timeout.
311  current = mp_nodes[previous].next;
312  if (current != TIMER_NULL)
313  {
314  mp_nodes[current].ticks_to_expire += timeout;
315  }
316 }

359 static void timer_timeouts_check(void)
360 {
361  // Handle expired of timer.
362  if (m_timer_id_head != TIMER_NULL)
363  {
364  uint32_t ticks_expired;
365  uint8_t ticks_elapsed_last;
366 
367  // Initialize actual elapsed ticks being consumed to 0.
368  ticks_expired = 0;
369 
370  // Queue the ticks elapsed (to make the value context safe).
371  ticks_elapsed_last = m_ticks_elapsed_last + 1;
372  if (ticks_elapsed_last == CONTEXT_QUEUE_SIZE_MAX)
373  {
374  ticks_elapsed_last = 0;
375  }
376 
377  if (ticks_elapsed_last != m_ticks_elapsed_first)
378  {
379  app_timer_id_t timer_id;
380  uint32_t ticks_elapsed;
381 
382  // Ticks_elapsed is collected here, job will use it.
383  ticks_elapsed = ticks_diff_get(rtc1_counter_get(), m_ticks_latest);
384 
385  // Auto variable containing the head of timers expiring.
386  timer_id = m_timer_id_head;
387 
388  // Expire all timers within ticks_elapsed and collect ticks_expired.
389  while (timer_id != TIMER_NULL)
390  {
391  timer_node_t * p_timer;
392 
393  // Auto variable for current timer node.
394  p_timer = &mp_nodes[timer_id];
395 
396  // Do nothing if timer did not expire.
397  if (ticks_elapsed < p_timer->ticks_to_expire)
398  {
399  break;
400  }
401 
402  // Decrement ticks_elapsed and collect expired ticks
403  ticks_elapsed -= p_timer->ticks_to_expire;
404  ticks_expired += p_timer->ticks_to_expire;
405 
406  // Move to next timer
407  timer_id = p_timer->next;
408 
409  // Execute Task
410  timeout_handler_exec(p_timer);
411  }
412  }
413 
414  // Queue the elapsed value
415  m_ticks_elapsed[m_ticks_elapsed_last] = ticks_expired;
416  m_ticks_elapsed_last = ticks_elapsed_last;
417 
418  timer_list_handler_sched();
419  }
420 }

429 static bool elapsed_ticks_acquire(uint32_t * p_ticks_elapsed)
430 {
431  // Pick the elapsed value from queue
432  if (m_ticks_elapsed_first != m_ticks_elapsed_last)
433  {
434  *p_ticks_elapsed = m_ticks_elapsed[m_ticks_elapsed_first];
435 
436  // Dequeue elapsed value.
437  m_ticks_elapsed_first++;
438  if (m_ticks_elapsed_first == CONTEXT_QUEUE_SIZE_MAX)
439  {
440  m_ticks_elapsed_first = 0;
441  }
442 
443  m_ticks_latest += *p_ticks_elapsed;
444  m_ticks_latest &= MAX_RTC_CNT;
445 
446  return true;
447  }
448  else
449  {
450  // No elapsed value in queue.
451  *p_ticks_elapsed = 0;
452  return false;
453  }
454 }

461 static bool list_deletions_handler(void)
462 {
463  app_timer_id_t timer_id_old_head;
464  uint8_t user_id;
465 
466  // Remember the old head, so as to decide if new compare needs to be set.
467  timer_id_old_head = m_timer_id_head;
468 
469  user_id = m_user_array_size;
470  while (user_id--)
471  {
472  timer_user_t * p_user = &mp_users[user_id];
473  uint8_t user_ops_first = p_user->first;
474 
475  while (user_ops_first != p_user->last)
476  {
477  timer_node_t * p_timer;
478  timer_user_op_t * p_user_op = &p_user->p_user_op_queue[user_ops_first];
479 
480  // Traverse to next operation in queue
481  user_ops_first++;
482  if (user_ops_first == p_user->user_op_queue_size)
483  {
484  user_ops_first = 0;
485  }
486 
487  switch (p_user_op->op_type)
488  {
489  case TIMER_USER_OP_TYPE_STOP:
490  // Delete node if timer is running.
491  p_timer = &mp_nodes[p_user_op->timer_id];
492  if (p_timer->is_running)
493  {
494  timer_list_remove(p_user_op->timer_id);
495  p_timer->is_running = false;
496  }
497  break;
498 
499  case TIMER_USER_OP_TYPE_STOP_ALL:
500  // Delete list of running timers, and mark all timers as not running.
501  while (m_timer_id_head != TIMER_NULL)
502  {
503  timer_node_t * p_head = &mp_nodes[m_timer_id_head];
504 
505  p_head->is_running = false;
506  m_timer_id_head = p_head->next;
507  }
508  break;
509 
510  default:
511  // No implementation needed.
512  break;
513  }
514  }
515  }
516 
517  // Detect change in head of the list.
518  return (m_timer_id_head != timer_id_old_head);
519 }

528 static void expired_timers_handler(uint32_t ticks_elapsed,
529  uint32_t ticks_previous,
530  app_timer_id_t * p_restart_list_head)
531 {
532  uint32_t ticks_expired = 0;
533 
534  while (m_timer_id_head != TIMER_NULL)
535  {
536  timer_node_t * p_timer;
537  app_timer_id_t id_expired;
538 
539  // Auto variable for current timer node.
540  p_timer = &mp_nodes[m_timer_id_head];
541 
542  // Do nothing if timer did not expire.
543  if (ticks_elapsed < p_timer->ticks_to_expire)
544  {
545  p_timer->ticks_to_expire -= ticks_elapsed;
546  break;
547  }
548 
549  // Decrement ticks_elapsed and collect expired ticks.
550  ticks_elapsed -= p_timer->ticks_to_expire;
551  ticks_expired += p_timer->ticks_to_expire;
552 
553  // Timer expired, set ticks_to_expire zero.
554  p_timer->ticks_to_expire = 0;
555  p_timer->is_running = false;
556 
557  // Remove the expired timer from head.
558  id_expired = m_timer_id_head;
559  m_timer_id_head = p_timer->next;
560 
561  // Timer will be restarted if periodic.
562  if (p_timer->ticks_periodic_interval != 0)
563  {
564  p_timer->ticks_at_start = (ticks_previous + ticks_expired) & MAX_RTC_CNT;
565  p_timer->ticks_first_interval = p_timer->ticks_periodic_interval;
566  p_timer->next = *p_restart_list_head;
567  *p_restart_list_head = id_expired;
568  }
569  }
570 }

579 static bool list_insertions_handler(app_timer_id_t restart_list_head)
580 {
581  app_timer_id_t timer_id_old_head;
582  uint8_t user_id;
583 
584  // Remember the old head, so as to decide if new compare needs to be set.
585  timer_id_old_head = m_timer_id_head;
586 
587  user_id = m_user_array_size;
588  while (user_id--)
589  {
590  timer_user_t * p_user = &mp_users[user_id];
591 
592  // Handle insertions of timers.
593  while ((restart_list_head != TIMER_NULL) || (p_user->first != p_user->last))
594  {
595  app_timer_id_t id_start;
596  timer_node_t * p_timer;
597 
598  if (restart_list_head != TIMER_NULL)
599  {
600  id_start = restart_list_head;
601  p_timer = &mp_nodes[id_start];
602  restart_list_head = p_timer->next;
603  }
604  else
605  {
606  timer_user_op_t * p_user_op = &p_user->p_user_op_queue[p_user->first];
607 
608  p_user->first++;
609  if (p_user->first == p_user->user_op_queue_size)
610  {
611  p_user->first = 0;
612  }
613 
614  id_start = p_user_op->timer_id;
615  p_timer = &mp_nodes[id_start];
616 
617  if ((p_user_op->op_type != TIMER_USER_OP_TYPE_START) || p_timer->is_running)
618  {
619  continue;
620  }
621 
622  p_timer->ticks_at_start = p_user_op->params.start.ticks_at_start;
623  p_timer->ticks_first_interval = p_user_op->params.start.ticks_first_interval;
624  p_timer->ticks_periodic_interval = p_user_op->params.start.ticks_periodic_interval;
625  p_timer->p_context = p_user_op->params.start.p_context;
626  }
627 
628  // Prepare the node to be inserted.
629  if (((p_timer->ticks_at_start - m_ticks_latest) & MAX_RTC_CNT) < (MAX_RTC_CNT / 2))
630  {
631  p_timer->ticks_to_expire = ticks_diff_get(p_timer->ticks_at_start, m_ticks_latest) +
632  p_timer->ticks_first_interval;
633  }
634  else
635  {
636  uint32_t delta_current_start;
637 
638  delta_current_start = ticks_diff_get(m_ticks_latest, p_timer->ticks_at_start);
639  if (p_timer->ticks_first_interval > delta_current_start)
640  {
641  p_timer->ticks_to_expire = p_timer->ticks_first_interval - delta_current_start;
642  }
643  else
644  {
645  p_timer->ticks_to_expire = 0;
646  }
647  }
648  p_timer->ticks_at_start = 0;
649  p_timer->ticks_first_interval = 0;
650  p_timer->is_running = true;
651  p_timer->next = TIMER_NULL;
652 
653  // Insert into list.
654  timer_list_insert(id_start);
655  }
656  }
657 
658  return (m_timer_id_head != timer_id_old_head);
659 }

664 static void compare_reg_update(app_timer_id_t timer_id_head_old)
665 {
666  // Setup the timeout for timers on the head of the list.
667  if (m_timer_id_head != TIMER_NULL)
668  {
669  uint32_t ticks_to_expire = mp_nodes[m_timer_id_head].ticks_to_expire;
670  uint32_t counter = rtc1_counter_get();
671  uint32_t cc = m_ticks_latest;
672  uint32_t ticks_elapsed = ticks_diff_get(counter, cc) + RTC_COMPARE_OFFSET_MIN;
673 
674  if (timer_id_head_old == TIMER_NULL)
675  {
676  // No timers were already running, start RTC.
677  rtc1_start();
678  }
679 
680  cc += (ticks_elapsed < ticks_to_expire) ? ticks_to_expire : ticks_elapsed;
681  cc &= MAX_RTC_CNT;
682 
683  rtc1_compare0_set(cc);
684 
685  if (ticks_diff_get(rtc1_counter_get(), counter) > ticks_diff_get(cc, counter))
686  {
687  timer_timeouts_check_sched();
688  }
689  }
690  else
691  {
692  // No timers are running, stop RTC.
693  rtc1_stop();
694  }
695 }

700 static void timer_list_handler(void)
701 {
702  app_timer_id_t restart_list_head = TIMER_NULL;
703  uint32_t ticks_elapsed;
704  uint32_t ticks_previous;
705  bool ticks_have_elapsed;
706  bool compare_update;
707  app_timer_id_t timer_id_head_old;
708 
709  // Back up the previous known tick and previous list head.
710  ticks_previous = m_ticks_latest;
711  timer_id_head_old = m_timer_id_head;
712 
713  // Get number of elapsed ticks.
714  ticks_have_elapsed = elapsed_ticks_acquire(&ticks_elapsed);
715 
716  // Handle list deletions.
717  compare_update = list_deletions_handler();
718 
719  // Handle expired timers.
720  if (ticks_have_elapsed)
721  {
722  expired_timers_handler(ticks_elapsed, ticks_previous, &restart_list_head);
723  compare_update = true;
724  }
725 
726  // Handle list insertions.
727  if (list_insertions_handler(restart_list_head))
728  {
729  compare_update = true;
730  }
731 
732  // Update compare register if necessary.
733  if (compare_update)
734  {
735  compare_reg_update(timer_id_head_old);
736  }
737 }

875 uint32_t app_timer_init(uint32_t prescaler,
876  uint8_t max_timers,
877  uint8_t op_queues_size,
878  void * p_buffer,
879  app_timer_evt_schedule_func_t evt_schedule_func)
880 {
881  int i;
882 
883  // Check that buffer is correctly aligned.
884  if (!is_word_aligned(p_buffer))
885  {
886  return NRF_ERROR_INVALID_PARAM;
887  }
888 
889  // Stop RTC to prevent any running timers from expiring (in case of reinitialization).
890  rtc1_stop();
891 
892  m_evt_schedule_func = evt_schedule_func;
893 
894  // Initialize timer node array.
895  m_node_array_size = max_timers;
896  mp_nodes = p_buffer;
897 
898  for (i = 0; i < max_timers; i++)
899  {
900  mp_nodes[i].state = STATE_FREE;
901  mp_nodes[i].is_running = false;
902  }
903 
904  // Skip timer node array.
905  p_buffer = &((uint8_t *)p_buffer)[max_timers * sizeof(timer_node_t)];
906 
907  // Initialize users array.
908  m_user_array_size = APP_TIMER_INT_LEVELS;
909  mp_users = p_buffer;
910 
911  // Skip user array.
912  p_buffer = &((uint8_t *)p_buffer)[APP_TIMER_INT_LEVELS * sizeof(timer_user_t)];
913 
914  // Initialize operation queues.
915  for (i = 0; i < APP_TIMER_INT_LEVELS; i++)
916  {
917  timer_user_t * p_user = &mp_users[i];
918 
919  p_user->first = 0;
920  p_user->last = 0;
921  p_user->user_op_queue_size = op_queues_size;
922  p_user->p_user_op_queue = p_buffer;
923 
924  // Skip operation queue.
925  p_buffer = &((uint8_t *)p_buffer)[op_queues_size * sizeof(timer_user_op_t)];
926  }
927 
928  m_timer_id_head = TIMER_NULL;
929  m_ticks_elapsed_first = 0;
930  m_ticks_elapsed_last = 0;
931 
932  NVIC_ClearPendingIRQ(SWI1_IRQn);
933  NVIC_SetPriority(SWI1_IRQn, SWI1_IRQ_PRI);
934  NVIC_EnableIRQ(SWI1_IRQn);
935 
936  rtc1_init(prescaler);
937 
938  m_ticks_latest = rtc1_counter_get();
939 
940  return NRF_SUCCESS;
941 }
