372 static void beacon_write_handler(ble_bcs_t * p_lbs, beacon_data_type_t type, uint8_t *data)
373 {
374  uint32_t err_code;
375 
376  static beacon_flash_db_t tmp;
377 
378  memcpy(&tmp, p_beacon, sizeof(beacon_flash_db_t));
379 
380  tmp.data.magic_byte = MAGIC_FLASH_BYTE;
381 
382  switch(type)
383  {
384  case beacon_maj_min_data:
385  tmp.data.beacon_data[BEACON_MANUF_DAT_MAJOR_H_IDX] = data[0];
386  tmp.data.beacon_data[BEACON_MANUF_DAT_MAJOR_L_IDX] = data[1];
387  tmp.data.beacon_data[BEACON_MANUF_DAT_MINOR_H_IDX] = data[2];
388  tmp.data.beacon_data[BEACON_MANUF_DAT_MINOR_L_IDX] = data[3];
389  break;
390 
391  case beacon_measured_rssi_data:
392  tmp.data.beacon_data[BEACON_MANUF_DAT_RSSI_IDX] = data[0];
393  break;
394 
395  case beacon_uuid_data:
396  memcpy(&tmp.data.beacon_data[BEACON_MANUF_DAT_UUID_IDX], data, BCS_DATA_ID_LEN);
397  break;
398 
399  case beacon_company_id_data:
400  tmp.data.company_id = (data[1] << 8) + data[0];
401  break;
402 
403  case beacon_adv_interval_data:
404  tmp.data.adv_interval = (data[1] << 8) + data[0];
405  if (tmp.data.adv_interval < APP_BEACON_ADV_INTERVAL_MIN_MS)
406  {
407  tmp.data.adv_interval = APP_BEACON_ADV_INTERVAL_MIN_MS;
408  }
409  else if (tmp.data.adv_interval > APP_BEACON_ADV_INTERVAL_MAX_MS)
410  {
411  tmp.data.adv_interval = APP_BEACON_ADV_INTERVAL_MAX_MS;
412  }
413  break;
414 
415  case beacon_led_data:
416  tmp.data.led_state = data[0];
417  break;
418 
419  default:
420  break;
421  }
422 
423  err_code = pstorage_clear(&m_pstorage_block_id, sizeof(beacon_flash_db_t));
424  APP_ERROR_CHECK(err_code);
425 
426  err_code = pstorage_store(&m_pstorage_block_id, (uint8_t *)&tmp, sizeof(beacon_flash_db_t), 0);
427  APP_ERROR_CHECK(err_code);
428 }

594 static void on_ble_evt(ble_evt_t * p_ble_evt)
595 {
596  uint32_t err_code = NRF_SUCCESS;
597  static ble_gap_evt_auth_status_t m_auth_status;
598  ble_gap_enc_info_t * p_enc_info;
599 
600  switch (p_ble_evt->header.evt_id)
601  {
602  case BLE_GAP_EVT_CONNECTED:
603  m_conn_handle = p_ble_evt->evt.gap_evt.conn_handle;
604  break;
605 
606  case BLE_GAP_EVT_DISCONNECTED:
607  m_conn_handle = BLE_CONN_HANDLE_INVALID;
608  beacon_reset();
609  break;
610 
611  case BLE_GAP_EVT_SEC_PARAMS_REQUEST:
612  err_code = sd_ble_gap_sec_params_reply(m_conn_handle,
613  BLE_GAP_SEC_STATUS_SUCCESS,
614  &m_sec_params);
615  break;
616 
617  case BLE_GATTS_EVT_SYS_ATTR_MISSING:
618  err_code = sd_ble_gatts_sys_attr_set(m_conn_handle, NULL, 0);
619  break;
620 
621  case BLE_GAP_EVT_AUTH_STATUS:
622  m_auth_status = p_ble_evt->evt.gap_evt.params.auth_status;
623  break;
624 
625  case BLE_GAP_EVT_SEC_INFO_REQUEST:
626  p_enc_info = &m_auth_status.periph_keys.enc_info;
627  if (p_enc_info->div == p_ble_evt->evt.gap_evt.params.sec_info_request.div)
628  {
629  err_code = sd_ble_gap_sec_info_reply(m_conn_handle, p_enc_info, NULL);
630  }
631  else
632  {
633  // No keys found for this device
634  err_code = sd_ble_gap_sec_info_reply(m_conn_handle, NULL, NULL);
635  }
636  break;
637 
638  case BLE_GAP_EVT_TIMEOUT:
639  if (p_ble_evt->evt.gap_evt.params.timeout.src == BLE_GAP_TIMEOUT_SRC_ADVERTISEMENT)
640  {
641  beacon_reset();
642  }
643  break;
644 
645  default:
646  break;
647  }
648 
649  APP_ERROR_CHECK(err_code);
650 }

734 static void beacon_params_default_set(void)
735 {

739  uint8_t beacon_data[APP_BEACON_MANUF_DATA_LEN] =

748  APP_DEFAULT_MEASURED_RSSI // Manufacturer specific information. The beacon's measured TX power in
749  // this implementation.
750  };
751 
752  tmp.data.magic_byte = MAGIC_FLASH_BYTE;
753  tmp.data.adv_interval = APP_BEACON_DEFAULT_ADV_INTERVAL_MS;
754  tmp.data.company_id = APP_DEFAULT_COMPANY_IDENTIFIER;
755  tmp.data.led_state = 0x01;
756 
757  beacon_data[BEACON_MANUF_DAT_MINOR_L_IDX] = (uint8_t)(NRF_FICR->DEVICEADDR[0] & 0xFFUL);
758  beacon_data[BEACON_MANUF_DAT_MINOR_H_IDX] = (uint8_t)((NRF_FICR->DEVICEADDR[0] >> 8) & 0xFFUL);
759  beacon_data[BEACON_MANUF_DAT_MAJOR_L_IDX] = (uint8_t)((NRF_FICR->DEVICEADDR[0] >> 16) & 0xFFUL);
760  beacon_data[BEACON_MANUF_DAT_MAJOR_H_IDX] = (uint8_t)((NRF_FICR->DEVICEADDR[0] >> 24) & 0xFFUL);
761 
762  memcpy(tmp.data.beacon_data, beacon_data, APP_BEACON_MANUF_DATA_LEN);

764  err_code = pstorage_clear(&m_pstorage_block_id, sizeof(beacon_flash_db_t));
765  APP_ERROR_CHECK(err_code);
766 
767  err_code = pstorage_store(&m_pstorage_block_id, (uint8_t *)&tmp, sizeof(beacon_flash_db_t), 0);
768  APP_ERROR_CHECK(err_code);
769 }
